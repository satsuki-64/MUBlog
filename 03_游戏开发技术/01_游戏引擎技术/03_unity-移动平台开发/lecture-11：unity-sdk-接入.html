在 Unity 中进行 SDK 接入基础概念

什么是 SDK 接入

概念：什么是 SDK 接入

  • 概念：
    • 在 Unity 移动平台开发中，SDK 是连接你的游戏与手机原生功能（如登录、支付）的桥梁；
    • SDK 当中包含很多 API，这些 API 是 SDK 的一部分，是 SDK 实现功能的核心手段。SDK 是一个完整的“工具箱”，而 API 是这个箱子里所有工具的“使用说明书”和“接口”；
  • 概念：Unity 接入第三方 SDK 的套路
    • 第一步：阅读官方提供的接入文档
      • 1-1 合作方提供；
      • 1-2 官方网站获取；
    • 第二步：按照接入文档中的内容进行操作
      • 2-1 依赖配置、权限配置；
      • 2-2 API 使用说明；
    • 第三步：接入过程中，根据报错解决问题
  • 注意：
    • 如果第三方针对 Unity 提供了 SDK，一般可以直接使用，不需要创建交互项目；
    • 如果没有提供，一般都是针对 Android 和 IOS 单独的支持，需要创建交互项目；

import com.baidu.location.BDAbstractLocationListener;
import com.baidu.location.BDLocation;
import com.baidu.location.LocationClient;
import com.baidu.location.LocationClientOption;
import com.baidu.location.LocationClientOption.LocationMode;

/**
 * 百度地图定位功能工具类
 * 用于在Unity中集成百度地图定位SDK
 */
public class BaiduLocationManager 
{
	
	private LocationClient mLocationClient;
	private LocationCallback mLocationCallback;
	
	/**
	 * 定位结果回调接口
	 */
	public interface LocationCallback {
		void onLocationSuccess(BDLocation location);
		void onLocationFailed(String errorMsg);
	}
	
	/**
	 * 初始化定位功能
	 * @param callback 定位结果回调
	 */
	public void initLocation(LocationCallback callback) {
		this.mLocationCallback = callback;
	}
	
	/**
	 * 开始获取地址定位信息
	 * 完整的定位流程方法
	 */
	public void GetAddr() {
		// ========== 第一步：隐私合规设置 ==========
		// setAgreePrivacy接口必须在LocationClient实例化之前调用
		// true：表示用户同意隐私合规政策，定位功能可以正常使用
		// false：表示用户不同意隐私合规政策，定位功能将无法实现
		LocationClient.setAgreePrivacy(true);
		
		try {
			// ========== 第二步：初始化定位客户端 ==========
			// 声明并实例化LocationClient类
			mLocationClient = new LocationClient(getApplicationContext());
			
			// ========== 第三步：配置定位参数 ==========
			LocationClientOption option = new LocationClientOption();
			
			// 设置定位模式（可选，默认高精度）
			// LocationMode.Hight_Accuracy：高精度模式，优先使用GPS，网络辅助定位，精度最高，耗电量大
			// LocationMode.Battery_Saving：低功耗模式，只使用网络定位（WiFi和基站），精度较低，省电
			// LocationMode.Device_Sensors：仅设备模式，只使用GPS进行定位，需要室外环境
			option.setLocationMode(LocationMode.Hight_Accuracy);
			
			// 设置返回经纬度坐标类型（可选，默认GCJ02）
			// "gcj02"：国测局坐标（火星坐标系），国内地图常用
			// "bd09ll"：百度经纬度坐标（BD0911），百度地图专用
			// "bd09"：百度墨卡托坐标
			// 注意：海外地区定位无需设置坐标类型，统一返回WGS84类型坐标
			option.setCoorType("bd09ll");
			
			// 设置是否需要地址信息，默认为无地址
			option.setIsNeedAddress(true);
			
			// 设置是否开启GPS定位（高精度模式下自动开启）
			option.setOpenGps(true);
			
			// 设置定位间隔（单位：毫秒）
			option.setScanSpan(1000);
			
			// 设置是否允许模拟位置
			option.setIgnoreKillProcess(false);
			
			// 将配置参数设置给定位客户端
			mLocationClient.setLocOption(option);
			
			// ========== 第四步：注册定位监听器 ==========
			// 创建自定义定位监听器实例
			MyLocationListener myListener = new MyLocationListener();
			mLocationClient.registerLocationListener(myListener);
			
			// ========== 第五步：开始定位 ==========
			// 启动定位客户端
			if (mLocationClient != null && !mLocationClient.isStarted()) {
				mLocationClient.start();
			}
			
		} catch (Exception e) {
			e.printStackTrace();
			if (mLocationCallback != null) {
				mLocationCallback.onLocationFailed("定位初始化失败: " + e.getMessage());
			}
		}
	}
	
	/**
	 * 停止定位
	 */
	public void stopLocation() {
		if (mLocationClient != null && mLocationClient.isStarted()) {
			mLocationClient.stop();
		}
	}
	
	/**
	 * 释放定位资源
	 */
	public void destroy() {
		if (mLocationClient != null) {
			mLocationClient.stop();
			mLocationClient = null;
		}
	}
	
	/**
	 * 自定义定位监听器
	 * 处理定位结果的回调
	 */
	public class MyLocationListener extends BDAbstractLocationListener {
		@Override
		public void onReceiveLocation(BDLocation location) {
			if (location == null) {
				if (mLocationCallback != null) {
					mLocationCallback.onLocationFailed("定位结果为空");
				}
				return;
			}
			
			// 获取定位错误码
			int locType = location.getLocType();
			
			switch (locType) {
				case BDLocation.TypeGpsLocation:
					// GPS定位结果
					handleLocationSuccess(location, "GPS定位成功");
					break;
					
				case BDLocation.TypeNetWorkLocation:
					// 网络定位结果
					handleLocationSuccess(location, "网络定位成功");
					break;
					
				case BDLocation.TypeOffLineLocation:
					// 离线定位结果
					handleLocationSuccess(location, "离线定位成功");
					break;
					
				case BDLocation.TypeServerError:
					// 服务端网络定位失败
					handleLocationFailed("服务端网络定位失败");
					break;
					
				case BDLocation.TypeNetWorkException:
					// 网络不同导致定位失败
					handleLocationFailed("网络异常导致定位失败");
					break;
					
				case BDLocation.TypeCriteriaException:
					// 无法获取有效定位依据导致定位失败
					handleLocationFailed("无法获取有效定位信息");
					break;
					
				default:
					handleLocationFailed("定位失败，错误码: " + locType);
					break;
			}
			
			// 停止定位（单次定位）
			stopLocation();
		}
		
		/**
		 * 处理定位成功结果
		 */
		private void handleLocationSuccess(BDLocation location, String msg) {
			// 获取经纬度
			double latitude = location.getLatitude();
			double longitude = location.getLongitude();
			
			// 获取地址信息
			String address = location.getAddrStr();
			String country = location.getCountry();
			String province = location.getProvince();
			String city = location.getCity();
			String district = location.getDistrict();
			String street = location.getStreet();
			
			if (mLocationCallback != null) {
				mLocationCallback.onLocationSuccess(location);
			}
		}
		
		/**
		 * 处理定位失败
		 */
		private void handleLocationFailed(String errorMsg) {
			if (mLocationCallback != null) {
				mLocationCallback.onLocationFailed(errorMsg);
			}
		}
	}
	
	/**
	 * Unity调用接口 - 开始定位
	 */
	public static void StartBaiduLocation() {
		BaiduLocationManager manager = new BaiduLocationManager();
		manager.GetAddr();
	}
	
	/**
	 * Unity调用接口 - 停止定位
	 */
	public static void StopBaiduLocation() {
		BaiduLocationManager manager = new BaiduLocationManager();
		manager.stopLocation();
	}
}

第二步：将相关信息传递至 Unity

概念：将相关信息传递至 Unity 

using System;
using UnityEngine;

/**
 * Unity端调用百度地图定位的C#脚本
 */
public class BaiduLocationController : MonoBehaviour
{
    private AndroidJavaObject baiduLocationPlugin;
    
    void Start()
    {
        InitializeBaiduLocation();
    }
    
    /// <summary>
    /// 初始化百度定位功能
    /// </summary>
    private void InitializeBaiduLocation()
    {
        try
        {
            // 通过AndroidJavaClass调用Java端的定位功能
            AndroidJavaClass unityPlayer = new AndroidJavaClass("com.unity3d.player.UnityPlayer");
            AndroidJavaObject currentActivity = unityPlayer.GetStatic<AndroidJavaObject>("currentActivity");
            
            // 创建类实例，并且调用“开始定位”方法
            AndroidJavaClass locationClass = new AndroidJavaClass("com.yourcompany.baidu.BaiduLocationManager");
            locationClass.CallStatic("StartBaiduLocation");
        }
        catch (Exception e)
        {
            Debug.LogError("百度定位初始化失败: " + e.Message);
        }
    }
    
    /// <summary>
    /// 开始定位
    /// </summary>
    public void StartLocation()
    {
        if (Application.platform == RuntimePlatform.Android)
        {
            InitializeBaiduLocation();
        }
        else
        {
            Debug.LogWarning("百度定位功能仅在Android平台可用");
        }
    }
    
    /// <summary>
    /// 停止定位
    /// </summary>
    public void StopLocation()
    {
        try
        {
            AndroidJavaClass locationClass = new AndroidJavaClass("com.yourcompany.baidu.BaiduLocationManager");
            locationClass.CallStatic("StopBaiduLocation");
        }
        catch (Exception e)
        {
            Debug.LogError("停止定位失败: " + e.Message);
        }
    }
}
