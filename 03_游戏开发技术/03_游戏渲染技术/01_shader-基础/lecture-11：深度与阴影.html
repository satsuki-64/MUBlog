Lecture 11：深度与阴影

深度缓冲区

基础概念

概念：深度缓冲区

  • 概念：深度缓冲区
    • 深度缓冲区是一个与屏幕分辨率相同的二维数组，每个像素对应一个深度值（通常为 24 位或 32 位浮点数），存储的是当前像素距离相机的远近​（在相机空间中的 Z 坐标）
    • 深度值范围是 0（近裁剪面）到 1（远裁剪面），但精度分布非线性的。近处精度高，远处精度低，这是透视投影的特性；
  • 概念：深度信息
    • 当经过 MVP 变换之后，3D 投影为了 2D，此时变换后的 Z 值就构成了深度信息；
    • 每个顶点的深度信息，在光栅化阶段会被插值，用于决定其顶点范围内的片元的深度值，并写入到深度缓冲中 (如果开启了深度写入的话)；

深度写入

概念：深度写入

  • 概念：
    • 深度写入 (ZWrite) 是 Unity Shader 中控制物体深度信息如何写入深度缓冲区的关键设置，它直接影响 3D 场景中物体的遮挡关系和渲染顺序；
    • 深度写入控制是否将物体的深度信息写入深度缓冲区，它只有两个状态：
        1. ZWrite On​：开启深度写入 (默认状态)；
        1. ZWrite Off​：关闭深度写入；
  • 分析：开启深度写入
    • 开启深度写入时，物体会将其深度值写入深度缓冲区，在复杂 3D 场景中维护正确的物体遮挡关系，避免过度绘制，后续物体可以通过深度测试提前丢弃不可见片段；
    • 因此不透明物体一定要把深度写入打开；
  • 分析：关闭深度写入
    • 关闭深度写入时，物体不会更新深度缓冲区，适用于透明/半透明物体​，这些物体需要与背景或其他物体进行颜色混合，因此如果开启深度写入，则效果会有问题；
  • 图示：关闭深入写入的半透明材质球
  • 分析：如果一个不透明的模型的结构复杂，希望可以透过它看到背景，但是又希望模型自身的穿透效果（顺序）正确
SubShader
{
	Tags{"Quene" = "Transparent"}
	Blend [_ScrBlend] [_DstBlend]
	Cull [_Cull]
	
	Pass
	{
		// 开启深度写入
		ZWrite On
		
		// 但是第一个通道里面不输出颜色
		ColorMask 0
	}
	
	Pass
	{
		// ... 正常执行其他着色器效果
	}
}

Shader "Unlit/Course_03"  
{  
    Properties  
    {  
        [Header(RenderingMode)]  
        [Enum(UnityEngine.Rendering.BlendMode)]_SrcBlend("SrcBlend",Int) = 0  
        [Enum(UnityEngine.Rendering.BlendMode)]_DestBlend("DestBlend",Int) = 0  
        [Enum(UnityEngine.Rendering.BlendMode)]_Cull("Cull",Int) = 0  
        [Enum(Off,0,On,1)]_ZWrite("ZWrite",Int) = 0  
        
        [Header(Base)]
        _MainTex ("Texture", 2D) = "white"{}  
        _Color("Color",Color) = (1,1,1,1)  
        _Intensity("Intensity",Range(-4,4)) = 1  
        _MainUVSpeedX("MainUVSpeedX",Float) = 0  
        _MainUVSpeedY("MainUVSpeedY",Float) = 0  
        
        [Header(Mask)]  
        [Toggle]_MaskEnable("MaskEnable",Int) = 0  
        _MaskTex("MaskTex",2D) = "white"{}  
        _MaskUVSpeedX("MaskUVSpeedX",Float) = 0  
        _MaskUVSpeedY("MaskUVSpeedY",Float) = 0  
        
        [Header(Distort)]  
        [MaterialToggle(DISTORTENABLED)]_DistortEnable("DistortEnable",Int) = 0  
        _DistortTex("DistortTex",2D) = "white"{}  
        _Distort("Distort",Int) = 1  
        _DistortUVSpeedX("DistortUVSpeedX",Float) = 0  
        _DistortUVSpeedY("DistortUVSpeedY",Float) = 0  
    }  
    SubShader  
    {  
        Tags { "RenderType"="Transparent" }  
        Blend [_SrcBlend] [_DestBlend]  
        Cull [_Cull]  
        ZWrite [_ZWrite]  
  
        Pass  
        {  
            CGPROGRAM  
            #pragma vertex vert  
            #pragma fragment frag  
            #pragma shader_feature _ _MASKENABLE_ON  
            #pragma shader_feature _ _DISTORTENABLE_ON  
            #pragma multi_compile_fog  
  
            #include "UnityCG.cginc"  
  
            struct appdata  
            {  
                float4 vertex : POSITION;  
                float2 uv : TEXCOORD0;  
            };  
            struct v2f  
            {  
                float2 uv : TEXCOORD0;  
                float2 uv2 : TEXCOORD1;  
                float2 uv3 : TEXCOORD2;  
                UNITY_FOG_COORDS(1)  
                float4 vertex : SV_POSITION;  
            };  
            
            sampler2D _MainTex; float4 _MainTex_ST;  
            float _MainUVSpeedX,_MainUVSpeedY;  
            sampler2D _MaskTex; float4 _MaskTex_ST;  
            float _MaskUVSpeedX,_MaskUVSpeedY;  
  
            sampler2D _DistortTex;float4 _DistortTex_ST;  
            float _Distort,_DistortUVSpeedX,_DistortUVSpeedY;  
            float4 _Color;  
            half _Intensity;  
            v2f vert (appdata v)  
            {                
	            v2f o;  
                o.vertex = UnityObjectToClipPos(v.vertex);  
                o.uv = TRANSFORM_TEX(v.uv, _MainTex) + float2(_MainUVSpeedX,_MainUVSpeedY)*_Time.y;  
                o.uv2 = TRANSFORM_TEX(v.uv,_DistortTex) + float2(_DistortUVSpeedX,_DistortUVSpeedY)*_Time.y;  
                o.uv3 = TRANSFORM_TEX(v.uv,_MaskTex) + float2(_MaskUVSpeedX,_MaskUVSpeedY)*_Time.y;  
				UNITY_TRANSFER_FOG(o,o.vertex);  
                return o;  
            }  
            fixed4 frag (v2f i) : SV_Target  
            {  
                fixed4 color;  
                color = _Color * _Intensity;                
                float2 distortValue = i.uv.xy;  
                fixed4 distortTex = tex2D(_DistortTex,i.uv2);  
                distortValue = lerp(i.uv.xy,distortTex.xy,_Distort);  
                fixed4 mainTex = tex2D(_MainTex,distortValue);  
                color *= mainTex;   
                fixed4 maskTex = tex2D(_MaskTex,i.uv3);  
                color *= maskTex;                
                UNITY_APPLY_FOG(i.fogCoord, col);  
                return color;  
            }            
            ENDCG  
        }  
    }
}

深度测试

概念：深度测试

  • 概念：
    • 深度测试 (Depth Test) 是图形渲染管线中的一个关键环节，它决定了哪些像素应该被绘制到屏幕上，哪些应该被丢弃；
    • 当渲染一个像素时，系统会比较当前像素的深度值与深度缓冲区中对应位置的存储值，根据比较结果决定是否更新颜色缓冲区和深度缓冲区；
  • 分类：
    • （1）标准不透明组合 (ZWrite On, ZTest LEqual)
      • 这是 Unity 中的默认配置，用于绝大多数不透明物体；
      • 工作原理：
        • 深度测试使用 LEqual（小于等于），意味着只有当片元比深度缓冲区中已有的值更近（或相等）时，才会通过测试并通过深度写入更新深度缓冲区；
        • 这确保了离相机最近的物体遮挡后面的物体，得到正确的遮挡关系；
      • 渲染顺序：
        • 为了充分利用 Early-Z 等硬件优化，提升渲染性能，不透明物体通常按从前往后​（离相机近的先渲染）的顺序渲染；
        • 先渲染的物体将深度写入深度缓冲区，后渲染的物体在其基础上的深度测试可以快速失败，避免不必要的片元着色器计算；
    • （2）透明效果组合 (ZWrite Off, ZTest LEqual)
      • 这是渲染半透明物体（如玻璃、粒子效果）最常用的组合；
      • 工作原理：
        • 深度测试仍然开启（LEqual），所以它不会被更远的不透明物体遮挡；
        • 但关键在于关闭了深度写入​（ZWrite Off），这意味着它不会破坏深度缓冲区中已有的深度信息，后续的物体（包括其他半透明物体）仍然能基于正确的深度进行测试；
      • 注意：
        • 使用了 ZWrite Off 的半透明物体，需要设置其颜色的混合，由 Blend 命令控制，比如 Blend SrcAlpha OneMinusSrcAlpha；
      • 渲染顺序：
        • 半透明物体通常按从后往前的顺序渲染，这是为了混合颜色时能得到正确的结果；
        • 先绘制远处的半透明物体，再绘制近处的，近处的颜色才能正确地与帧缓冲区中已有的（远处的）颜色进行混合；
      • 常用于玻璃、水面、粒子特效、烟雾等所有需要表现透光性和混合效果的物体；
    • （3）遮挡显示组合 (ZTest Greater, ZWrite On/Off)
      • 此组合用于实现诸如“透视”或“X 射线”效果，显示被遮挡的内容；
      • 工作原理​：
        • 将深度测试条件设置为 Greater，意味着只有当前片元比深度缓冲区中记录的值更远时才会通过测试并显示；
        • 它通常用于一个额外的 Pass 中，例如在角色被墙壁遮挡时，用一个 ZTest Greater 的 Pass 来绘制描边或高亮显示角色轮廓；
        • 此时深度写入（ZWrite）通常关闭，以免干扰后续主要物体的深度测试；
      • 常用于游戏中的透视技能、角色被遮挡时的轮廓提示（X-Ray 效果）、特殊高亮提示；
    • （4）全屏覆盖组合 (ZWrite Off, ZTest Always)
      • 此组合完全无视深度信息，确保物体绘制在最上层；
      • 工作原理​：
        • ZTest Always（或 ZTest Off）意味着无条件通过所有深度测试；
        • 同时关闭深度写入（ZWrite Off），避免污染深度缓冲区。这个组合使得物体的渲染完全依赖于其渲染队列（Queue）​​ 来决定谁在上层，通常需要设置较大的队列值；
      • 常用于 UI 元素、屏幕空间贴图、后期处理效果、镜头光晕等需要始终显示在最前面的内容；
    • （5）深度写入替代组合 (ZWrite On, ZTest Always)
      • 这是一个相对特殊的组合，用于精确控制深度缓冲区的写入；
      • 工作原理​：
        • ZTest Always 保证所有片元都通过深度测试，ZWrite On 则强制将当前物体的深度值写入深度缓冲区，​覆盖掉该像素位置上先前存储的深度值；
        • 这可以用于手动预填充深度缓冲区，改变后续物体深度测试的基准；
      • 常用于需要特别精确控制复杂透明物体渲染顺序时，可用一个 Pass 专门写入深度；某些特殊的遮罩或模板技术中，用于准备深度信息。需谨慎使用，因为它会破坏正常的深度关系；
  • 概念：ZTest

实例：XRay 效果

Shader "Unlit/Course_04"  
{  
    Properties  
    {  
        _Color("Color",Color) = (1,1,1,1)  
        _Fresnel("Fresnel",Int) = 2  
    }  
    
    SubShader  
    {  
        Tags  
        {  
            "RenderType"="Transparent"  
        }  
        
        ZTest LEqual  
        Blend One One   
        
        Pass        
        {  
            CGPROGRAM  
            #pragma vertex vert  
            #pragma fragment frag  
            #include "UnityCG.cginc"  
  
            struct appdata  
            {  
                float4 vertex : POSITION;  
                float2 uv : TEXCOORD0;  
                half3 normal : NORMAL;  
            };  
            
            struct v2f  
            {  
                float2 uv : TEXCOORD0;  
                float3 worldPos : TEXCOORD1;  
                half3 worldNormal : TEXCOORD2;  
                float4 vertex : SV_POSITION;  
            };
              
            sampler2D _MainTex;  
            float4 _MainTex_ST;  
            float4 _Color;  
            int _Fresnel;  
  
            v2f vert(appdata v)  
            {                
	            v2f o;  
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);  
                o.vertex = UnityObjectToClipPos(v.vertex);  
                o.worldPos = mul(unity_ObjectToWorld,v.vertex);  
                o.worldNormal = UnityObjectToWorldNormal(v.normal);  
                return o;  
            }  
            
            fixed4 frag(v2f i) : SV_Target  
            {  
                fixed4 c = 1;  
                
                // 获得一个从 当前顶点世界坐标 -> 相机坐标 的向量  
                // 注意：向量一般都要归一化  
                fixed3 V = normalize(_WorldSpaceCameraPos - i.worldPos);
                
				// 当前顶点的法线  
                fixed3 N = normalize(i.worldNormal);  
                
                // 点积操作，夹角越大、值越小，夹角越小、值越大  
                fixed VDotN = dot(V,N);  
                fixed4 fresnel = _Fresnel * pow(1-VDotN,_Fresnel);  
                c.rgb = fresnel * _Color.rgb;  
  
                fixed4 fracY = frac(i.worldPos.y * 20 - _Time.y);  
                c.rgb *= fracY;  
                return c;  
            }            
            ENDCG  
        }  
    }
}

Pass 的复用

概念：Pass 的复用

  • 概念：
    • 通过 UsePass "Shader路径/PASS_NAME" 直接引用其他 Shader 中已定义的 Pass，其中 PASS_NAME 必须转换为全大写形式；
  • 示例：
    • UsePass "Custom/MyShader/MYPASS" // 引用名为"MyPass"的Pass
  • 概念：命名规则
    • 被复用的 Pass 需通过 Name "PassName" 显式命名，否则无法被引用；
    • 命名后的 Pass 在 UsePass 中需全大写，例如 Name "NormalPass" 对应的 UsePass 为 ".../NORMALPASS"；

深度偏移

概念：深度偏移

  • 概念：深度信息
    • 深度缓冲区存储的是归一化设备坐标（NDC）的深度值，其范围因图形 API 而异：
      1. DirectX 平台​：NDC 的 z 范围为 [0, 1]，近裁剪面（Near）对应 0，远裁剪面（Far）对应 1；
      1. OpenGL：NDC 的 z 范围为 [-1, 1]，近裁剪面为 -1，远裁剪面为 1；
    • Unity 通过宏 UNITY_REVERSED_Z 适配不同平台，在 DX11/Metal 等平台上会反转深度方向（近裁剪面为 1，远裁剪面为 0）以提升精度；
  • 分析：深度值的非线性分布
    • 在 Camera 的近裁剪面附近，深度值变化快，精度高（如 z=10 时精度达 0.0001）；
    • 在 Camera 的远裁剪面附近，深度值变化慢，精度低（如 z=1000 时精度仅 0.01）；
  • 概念：深度偏移
    • 作用：
      • 通过 Offset Factor, Units 命令动态调整深度值，解决 Z 冲突（Z-Fighting）的问题；
    • 实现：
      • 在某个 Pass 或者 SubShader 当中写入 Offset -1,-1；
      • 其第一个参数表示 Factor，其基于多边形斜率（Slope）的动态偏移，陡峭表面偏移更大；
      • 第二个参数表示 Units，表示固定偏移量，与斜率无关；
      • 其计算公式为：
        • 最终深度偏移量 = Factor × max_slope + Units × min_depth_unit
    • 意义：
      • 可以简单理解，如果给的是负值，则类似于将模型的深度值推的靠近近平面。反之正值则是靠近远平面；

ShadowMapping 的原理

概念：ShadowMapping 的原理

  • 概念：
    • 在 Unity Shader 中，​Shadow Mapping 是实现动态阴影的核心技术，其通过深度比较判断遮挡关系，兼具效率与通用性；
    • 其相比光线追踪阴影，仅需额外渲染一次深度图，适合游戏等实时应用，并且可适配平行光、点光源、聚光灯等不同类型光源的阴影投射；
  • 概念：两阶段流程
      1. 深度图生成：从光源视角渲染场景，生成记录最小深度的纹理（Shadow Map），平行光用正交投影，点光源/聚光灯用透视投影；
      1. 阴影判定：从摄像机视角渲染时，将像素转换到光源空间，比较其深度与 Shadow Map 中的值；
      • 若当前的深度 > Shadow Map 对应值 -> 被遮挡；
      • 若当前的深度 < Shadow Map 对应值 -> 收到光照；
    • 其中 ShadowMap 是动态生成的阴影纹理图，在需要时对其进行采样来获取阴影的一种技术手段；
  • 实现：ShadowMapping 原理
      1. 在光源处添加一个相机，同时打开深度测试与写入，并生成 ShadowMap；
      1. 把当前片断的转换到光源空间下，计算得到的深度与采样正常绘制场景，ShadowMap 中的深度值进行比较，如果大于则表示该片断处于阴影中，反之则不在阴影中；
  • 分析：Unity 当中的实现 - 级联阴影映射（Cascaded Shadow Maps, CSM）
    • 分析：
      • 图示是主光源的深度图，平行光需要覆盖从近处到远处的整个场景，但单一 Shadow Map 无法兼顾所有区域的精度，因此使用四张深度图对应四个不同距离范围的级联层级（Cascade Levels），每层独立渲染不同精度的阴影；
      • 根据摄像机近/远裁剪面，将视锥体按距离划分为多个层级（如 0-10m、10-50m、50-200m、200-∞），对每个层级，从平行光视角渲染一张深度图（Shadow Map），存储该区域的最小深度值。渲染场景时，根据像素距离摄像机的远近，选择对应级联的深度图进行阴影计算；

阴影的投射

概念：阴影的投射

  • 概念：阴影投射的作用
    • 一个物体若要投射阴影，其 Shader 中就必须包含一个专门用于生成这张深度图的 Pass，即 LightMode 为 ShadowCaster 的 Pass，并且物体的 MeshRenderer 组件上需要开启 Cast Shadows 选项；
    • 而一个物体若要接收阴影，则其 MeshRenderer 组件上需要开启 Receive Shadows 选项，并在着色器中采样阴影信息；
  • 概念：生成阴影
    • 在正常的 Pass 之外，新写一个 Pass，并进行以下设置：
      1. 添加 "LightMope"="ShadowCaster" 的 Pass；
      • 这个标签告诉 Unity：这个 Pass 不用于常规渲染，而是专门用于为光源生成阴影贴图的，当 Unity 需要为场景中开启了“Cast Shadows”的物体生成阴影时，就会来调用这个 Pass；
      1. appdata 当中声明 float4 vertex : POSITION 和 half3 normal : NORMAL；
      1. 在 v2f 当中添加 V2F_SHADOW_CASTER 用于声明需要传输到片段着色器的数据；
      • V2F_SHADOW_CASTER 是一个 Unity 内置宏，它会自动展开为一个包含阴影投射计算所需数据的结构体，例如裁剪空间的位置等，只需声明即可；
      1. 在顶点着色器当中添加 TRANSFER_SHADOW_CASTER_NORMALOFFSET(0)，主要是计算阴影的偏移以解决不正确的 Shadow Acne 和 Peter Panning 现象；
      • 它主要完成两项工作：
      • （1）将顶点从模型空间变换到光源的裁剪空间；
      • （2）应用偏移（Bias）​​：这是解决阴影失真（Shadow Acne）和彼得潘现象（Peter Panning）的关键；
      • 该宏会巧妙地利用顶点法线（normal）信息来智能地施加一个偏移量，从而有效规避这些问题；
      1. 在片段着色器当中添加 SHADOW_CASTER_FRAGMENT(i)；
      • SHADOW_CASTER_FRAGMENT(i) 宏会输出该片元最终的深度值到 Shadow Map 中，这就是此 Pass 的最终目的，即生成一张深度图；
      1. 添加 #pragma multi_compile_shadowcaster；
      • 定义在 "LightMope"="ShadowCaster" 时，会自动生成两个宏：
      • （1）SHADOWS_DEPTH：用于生成直线光和聚光灯阴影；
      • （2）SHADOW_CUBE：用于生成点光源阴影；
  • 图示：
  • 补充：
      1. TRANSFER_SHADOW_CASTER_NORMALOFFSET：
      • 将顶点从模型空间转换到光源裁剪空间，并应用法线偏移（Normal Offset）来避免阴影痤疮（Shadow Acne）；
      • 此方法的内部会调用 UnityClipSpaceShadowCasterPos 方法，将顶点坐标转换到光源裁剪空间，并应用投影矩阵；
      • 并且通过 UnityApplyLinearShadowBias 调整裁剪空间 Z 值，避免自阴影问题；
      1. SHADOW_CASTER_FRAGMENT：
      • 生成阴影映射纹理（Shadow Map）所需的深度或距离值，具体行为取决于光源类型；
      • 对于平行光或者聚光灯而言，直接输出裁剪空间的深度值；
      • 而对于点光源而言，则输出计算顶点到光源的距离并编码为 CubeMap 格式；
      1. #pragma multi_compile_shadowcaster：
      • 通过此预编译命令，生成适用于不同光源类型（平行光、点光源、聚光灯）和渲染路径（正向/延迟渲染）的 Shader 变体；
      • Unity 编译时会展开以下预定义宏组合：

Shader "Custom/DissolveWithShadowCaster"
{
    Properties
    {
        // 溶解效果纹理

        // 溶解阈值控制

    }

    SubShader
    {
        Tags { "RenderType"="Opaque" "Queue"="Geometry" }

        // 常规渲染Pass
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            // 溶解效果开关
            #pragma multi_compile _ _DISSOLVEENABLED_ON
            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct v2f
            {
                float2 uv : TEXCOORD0;
                float4 pos : SV_POSITION;
            };

            float4 _DissolveTex_ST;
            fixed _Clip;

            }

            {

                half3 normal : NORMAL;
                float4 uv : TEXCOORD0;

			// 2. 在v2f中写入 V2F_SHADOW_CASTER

阴影的接收

概念：阴影的接收

  • 概念：采样阴影
    • 阴影采样发生在光照计算阶段​，通常是 Forward Base 或 ForwardAdd Pass，而非阴影贴图生成阶段 ShadowCaster Pass；
    • 具体而言，在顶点着色器当中使用 TRANSFER_SHADOW 宏将顶点坐标转换到光源空间，生成阴影采样坐标；
    • 然后，再在片元着色器当中，通过 UNITY_LIGHT_ATTENUATION 或 SHADOW_ATTENUATION 宏采样阴影贴图，比较深度值以确定阴影衰减因子；
  • 实现：
      1. 在 v2f 当中添加 UNITY_SHADOW_COORDS(idx)，unity 会自动声明一个叫做 _ShadowCoord 的 float4 变量，用作阴影的采样坐标；
      1. 在顶点着色器当中添加 TRANSFER_SHADOW(o)，用于将上面定义的 _ShadowCoord 纹理坐标变换到相应的屏幕空间纹理坐标，为采样阴影纹理使用；
      1. 在片段着色器当中添加 UNITY_LIGHT_ATTENUATION(atten,i,i.worldPos)，其中 atten 存储了采样后的阴影；
  • 代码：
Shader "Custom/ShadowDemo"
{
    Properties
    {
        _MainTex ("Main Texture", 2D) = "white" {}
        _Color ("Color", Color) = (1,1,1,1)
    }

    SubShader
    {
        Tags { "RenderType"="Opaque" "Queue"="Geometry" }

        // --- 主渲染Pass（处理光照+阴影接收）---
        Pass
        {
            Tags { "LightMode" = "ForwardBase" }
            
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            // 1. 生成主光源阴影接收变体
            #pragma multi_compile_fwdbase   
            #include "UnityCG.cginc"
            #include "Lighting.cginc"
            
            // 2. 引用阴影相关的宏
            // 阴影宏定义在此文件中
            #include "AutoLight.cginc"     

            struct appdata
            {
                float4 vertex : POSITION;
                float3 normal : NORMAL;
                float2 uv : TEXCOORD0;
            };

            struct v2f
            {
                float4 pos : SV_POSITION;
                float2 uv : TEXCOORD0;
                float3 worldNormal : TEXCOORD1;
                float3 worldPos : TEXCOORD2;
                
                // 3. 声明阴影坐标，其用于在阴影贴图上采样
                SHADOW_COORDS(3)          
            };

            sampler2D _MainTex;
            float4 _MainTex_ST;
            fixed4 _Color;

            v2f vert (appdata v)
            {
                v2f o;
                o.pos = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);
                o.worldNormal = UnityObjectToWorldNormal(v.normal);
                o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
                
                // 4. 计算阴影采样坐标
                TRANSFER_SHADOW(o)        
                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 texColor = tex2D(_MainTex, i.uv) * _Color;
                
                // 基础光照计算（漫反射）
                float3 lightDir = normalize(_WorldSpaceLightPos0.xyz);
                float3 normal = normalize(i.worldNormal);
                fixed3 diffuse = _LightColor0.rgb * texColor.rgb * max(0, dot(lightDir, normal));
                
                // 5. 阴影采样，其中 atten 包含阴影衰减因子
                UNITY_LIGHT_ATTENUATION(atten, i, i.worldPos) 
                
                // 6. 应用环境光+阴影衰减后的光照
                fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.rgb * texColor.rgb;
                fixed3 finalColor = ambient + diffuse * atten;
                
                return fixed4(finalColor, 1);
            }
            ENDCG
        }

        // --- 阴影投射Pass ---
        Pass
        {
            Tags { "LightMode" = "ShadowCaster" }
            
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_shadowcaster  // 生成不同光源类型的阴影变体
            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
                float3 normal : NORMAL;
            };

            struct v2f
            {
                V2F_SHADOW_CASTER;        // 内置阴影投射数据结构
            };

            v2f vert (appdata v)
            {
                v2f o;
                TRANSFER_SHADOW_CASTER_NORMALOFFSET(o)  // 坐标变换+法线偏移
                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {
                SHADOW_CASTER_FRAGMENT(i)  // 输出深度到ShadowMap
            }
            ENDCG
        }
    }
    FallBack "Diffuse"
}

ShaderLOD

概念：ShaderLOD

  • 概念：
    • ShaderLOD 是 Unity 中控制着色器计算复杂度的分级系统，其核心思想是根据硬件性能或视觉需求，选择不同精细度的着色器版本；
    • 与模型的 LOD 减少几何复杂度不同，ShaderLOD 通过简化着色计算来优化 GPU 性能，特别适用于需要适配多种硬件设备的项目；
    • ShaderLOD 是作用与 SubShader 上的，比如在 SubShader 当中写上 LOD 200；
  • 分析：ShaderLOD 的作用
    • ShaderLOD 是一种典型的"以空间换时间"的优化策略，通过预先准备多套着色器变体（增加内存占用）来换取运行时性能提升（减少 GPU 计算负担），这种技术在移动端和大型开放世界项目中尤为重要，可以显著降低高端效果在低端设备上运行导致的卡顿问题；
  • 概念：ShaderLOD 的值
    • 在 Unity 的 ShaderLab 语法中，每个 SubShader 都可以通过 LOD [数值] 命令指定一个细节层级值，取值范围通常在 1-1000 之间，数值越大表示复杂度越高；
    • 假设当前游戏设置高配置下 LOD 为 600，则在当前 Shader 当中、会往下寻找，寻找当前值当中小于等于 600 的第一个 SubShader 进行执行；
    • 即运行时系统会根据以下规则选择适当的 SubShader：
        1. Unity 优先选择 LOD 值小于等于当前材质 LOD 等级且硬件支持的第一个 SubShader；
        1. 若材质的 LOD 属性为 N，则跳过所有 LOD 值 >N 的 SubShader；
        1. 通过 Shader.globalMaximumLOD 可强制所有着色器使用不超过该值的 SubShader；
    • 一般写 SubShader 时，LOD 从高往低来写；

模型网格阴影

模型网格阴影原理

概念：模型网格阴影原理

  • 概念：模型网格阴影的原理
    • 网格阴影 (Grid-based Shadows) 是一种基于空间离散化的阴影生成技术，其核心思想是将 3D 场景投影到 2D 阴影平面上，并将该平面划分为规则的网格结构。然后对此阴影进行偏移，靠模型较远的区域偏移的更大，越靠近模型的区域偏移的越小；
  • 特点：
    • 优点：
        1. 实现简单；
        1. 效果实时，清晰；
    • 缺点：
        1. 需要两个 Pass，第一个 Pass 是角色正常效果的绘制，第二个 Pass 是阴影独立的绘制；
        1. 在地面起伏较大时容易穿帮；

Pass
{
	// 半透明混合效果
	Blend SrcAlpha OneMinusSrcAlpha
	
	// 确保渲染出来的阴影效果不是模型压缩后画面的重叠，因此进行模板测试
	Stencil
	{
		Ref 100
		Comp NotEqual
		Pass Replace
	}
	
	CGPROGRAM
	#pragma vertex vert
	#pragma fragment frag

	#include "UnityCG.cginc"

	struct appdata
	{
		float4 vertex : POSITION;
	};

	struct v2f
	{
		float4 pos : SV_POSITION;
	};

	float _Shadow;
	float4 _ShadowOffset;
	float _ShadowAlpha;
	
	v2f vert(appdata v)
	{
		v2f o;
		
		// 第一步：将模型网格压缩到Y轴上的一个阴影面
		// 因此需要先获得模型在世界空间下的坐标
		float4 worldPos = mul(unity_ObjectToWorld, v.vertex);
		float worldPosY = v.vertex.y;
		worldPos.y = _Shadow;

		// 第二步：对模型压缩得到的阴影进行偏移
		// 通过将偏移值乘以"worldPosY-_Shadow"，使得当前阴影的偏移距离是根据模型距离地面的高度正相关，离地面越远、阴影偏移越大
		worldPos.xz += _ShadowOffset.xy * (worldPosY-_Shadow);

		// 当前的worldPos已经是经过了模型变化的结果，即MVP变换当中的M，因此接下来只需要执行VP变换即可
		o.pos = mul(UNITY_MATRIX_VP,worldPos);
		
		return o;
	}

	fixed4 frag(v2f i) : SV_Target
	{
		// 实现主要颜色是黑色，但通过alpha控制透明度
		fixed4 c = 0;
		c.a = _ShadowAlpha;
		
		return c;
	}
	
	ENDCG
}
