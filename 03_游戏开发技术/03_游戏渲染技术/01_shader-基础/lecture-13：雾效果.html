雾效的原理

基础概念

概念：雾效的原理

  • 概念：
    • 雾效的本质是模拟光线在空气中传播时的衰减现象，其数学基础是光线散射公式​：
    • 最终颜色物体颜色衰减因子雾颜色衰减因子
    • 其中的衰减因子（Fog Factor）​​：由物体到摄像机的距离（z）和雾的密度（density）决定，范围在 [0, 1] 之间；
  • 概念：雾效的混合公式
    • 最终的颜色雾效颜色物体颜色雾效混合因子
    • 其中混合因子收到两个方面的影响：（1）物体离视角的距离；（2）雾本身的浓度；

雾效的实现

雾效的实现：方法一

概念：雾效的实现 - 方法一

  • 概念：Unity 内部定义的雾相关变量
// 位于一个名为`UnityFog`的常量缓冲区（CBUFFER）中
CBUFFER_START(UnityFog)
	// 雾效的颜色，类型为 fixed4（RGBA格式）。这是雾效混合时的基础颜色，会与物体颜色进行插值
    fixed4 unity_FogColor;
    
    // 以下四个float4类型的分量存储在unity_FogParams全局变量中
    // 其存储了不同雾效模式（Linear、Exp、Exp2）所需的计算系数。其四个分量（x、y、z、w）分别对应不同雾效模式的参数
    // x = density / sqrt(ln(2)), useful for Exp2 mode
    // y = density / ln(2), useful for Exp mode
    // z = -1/(end-start), useful for Linear mode
    // w = end/(end-start), useful for Linear mode
    float4 unity_FogParams;
CBUFFER_END
  • 代码：完整代码实现 - 使用三种不同的衰减方式
Shader "Unlit/Course_09_Fog"
{
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                float fogFactor : TEXCOORD0;
            };


            v2f vert (appdata v)
            {
                v2f o;
                
                o.vertex = UnityObjectToClipPos(v.vertex);
                
                // 计算雾的衰减公式当中的 Z 值，其含义表示从相机到雾的距离
                float3 worldPos = mul(unity_ObjectToWorld,v.vertex);
                
                // length函数用于计算向量的模长
                float z = length(_WorldSpaceCameraPos - worldPos);
                
                #if defined(FOG_LINEAR)
                    // 雾的衰减公式 - 线性公式：(end - z) / (end - start) = z * (-1/(end-start)) + (end/(end-start))
                    o.fogFactor = z * unity_FogParams.z + unity_FogParams.w;
                #elif defined(FOG_EXP)
                    o.fogFactor = exp2(-unity_FogParams.y*z);
                #elif defined(FOG_EXP2)
                    float density = unity_FogParams.x*z;
                    o.fogFactor = exp2(-density*density);
                #endif
                
                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 c = 1;
                
                #if defined(FOG_LINEAR) || defined(FOG_EXP) || defined(FOG_EXP2)
                    // 注意：这里进行插值时，雾的颜色unity_FogColor需要放在前面，而当前物体的颜色c需要放在后面
                    // 因为距离越近、雾的颜色影响越小，而距离越远、雾的颜色影响越大
                    c = lerp(unity_FogColor,c,i.fogFactor);
                #endif
                
                return c;
            }
            ENDCG
        }
    }
}

雾效的实现：方法二

概念：雾效的实现 - 方法二

  • 概念：实现步骤
      1. 定义 ​#pragma multi_compile_fog​
      • 声明雾效所需的变体（FOG_LINEAR、FOG_EXP、FOG_EXP2），根据渲染设置自动编译对应雾模式的 Shader 变体；
      1. 定义雾效插值器 ​UNITY_FOG_COORDS(idx)​
      • 在 v2f 结构体中声明，定义一个雾效插值器，如 float fogCoord : TEXCOORDx；
      • 其用于从顶点着色器向片元着色器传递雾效系数（fogCoord）；
      1. 计算雾效强度 UNITY_TRANSFER_FOG(o, o.vertex)​
      • 根据顶点在裁剪空间的坐标（o.vertex）计算雾效强度，并存入 o.fogCoord，其内部使用利用 unity_FogParams 中的参数（如线性雾的 z 和 w 分量）计算距离衰减；
      1. 应用雾效果 - ​UNITY_APPLY_FOG(i.fogCoord, col)​
      • 将雾颜色（unity_FogColor）与物体颜色（col）按雾效系数混合：
  • 分析：选择不同的计算方法
    • 在以下代码当中，主要是根据当前所使用的不同平台、为其选择不同的雾效因子计算方法；
    • 比如如果定义了 SHADER_API_MOBILE 则表示移动平台，此时的 UNITY_TRANSFER_FOG 函数将会调用 UNITY_CALC_FOG_FACTOR 来进行雾效的计算；
// 检查是否启用了线性雾(FOG_LINEAR)、指数雾(FOG_EXP)或指数平方雾(FOG_EXP2)
#if defined(FOG_LINEAR) || defined(FOG_EXP) || defined(FOG_EXP2)
    // 定义雾效插值器，idx为插值器索引，使用float1类型存储雾效系数
    #define UNITY_FOG_COORDS(idx) UNITY_FOG_COORDS_PACKED(idx, float1)

    // 针对移动设备或Shader Model 2.0平台（低性能设备）
    #if (SHADER_TARGET < 30) || defined(SHADER_API_MOBILE)
        // 在顶点着色器中计算雾效因子
        // 使用UNITY_CALC_FOG_FACTOR函数计算得到雾效因子unityFogFactor
        // 将unityFogFactor存储到fogCoord.x
        #define UNITY_TRANSFER_FOG(o,outpos) UNITY_CALC_FOG_FACTOR((outpos).z); o.fogCoord.x = unityFogFactor
        
        // 将雾效因子与切线空间数据复用存储（节省插值器）
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_TSPACE(o,outpos) UNITY_CALC_FOG_FACTOR((outpos).z); o.tSpace1.y = tangentSign; o.tSpace2.y = unityFogFactor
        
        // 将雾效因子与世界坐标复用存储（worldPos.w）
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_WORLD_POS(o,outpos) UNITY_CALC_FOG_FACTOR((outpos).z); o.worldPos.w = unityFogFactor
        
        // 将雾效因子与视线向量复用存储（eyeVec.w）
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_EYE_VEC(o,outpos) UNITY_CALC_FOG_FACTOR((outpos).z); o.eyeVec.w = unityFogFactor
    
    #else
        // 针对Shader Model 3.0及以上平台（PC/主机/高性能设备）
        // 仅传递裁剪空间深度值（outpos.z），片元着色器中再计算雾效因子（每像素计算）
        #define UNITY_TRANSFER_FOG(o,outpos) o.fogCoord.x = (outpos).z
        // 复用切线空间、世界坐标或视线向量的w分量存储深度值
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_TSPACE(o,outpos) o.tSpace2.y = (outpos).z
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_WORLD_POS(o,outpos) o.worldPos.w = (outpos).z
        #define UNITY_TRANSFER_FOG_COMBINED_WITH_EYE_VEC(o,outpos) o.eyeVec.w = (outpos).z
    #endif
#else
    // 未启用任何雾效模式时，宏定义为空（避免无效代码）
    #define UNITY_FOG_COORDS(idx)
    #define UNITY_TRANSFER_FOG(o,outpos)
    #define UNITY_TRANSFER_FOG_COMBINED_WITH_TSPACE(o,outpos)
    #define UNITY_TRANSFER_FOG_COMBINED_WITH_WORLD_POS(o,outpos)
    #define UNITY_TRANSFER_FOG_COMBINED_WITH_EYE_VEC(o,outpos)
#endif
  • 分析：雾效因子计算
// 根据雾效模式选择不同的计算公式
#if defined(FOG_LINEAR)
    // 线性雾公式：(end-z)/(end-start) = z * (-1/(end-start)) + (end/(end-start))
    // 通过预计算的unity_FogParams.z和w优化计算（z=-1/(end-start), w=end/(end-start))
    #define UNITY_CALC_FOG_FACTOR_RAW(coord) float unityFogFactor = (coord) * unity_FogParams.z + unity_FogParams.w
    
#elif defined(FOG_EXP)
    // 指数雾公式：exp(-density*z)
    // unity_FogParams.y存储预计算的density/ln(2)，通过exp2优化指数计算
    #define UNITY_CALC_FOG_FACTOR_RAW(coord) float unityFogFactor = unity_FogParams.y * (coord); unityFogFactor = exp2(-unityFogFactor)
    
#elif defined(FOG_EXP2)
    // 指数平方雾公式：exp(-(density*z)^2)
    // unity_FogParams.x存储density/sqrt(ln(2))，通过平方运算增强衰减效果
    #define UNITY_CALC_FOG_FACTOR_RAW(coord) float unityFogFactor = unity_FogParams.x * (coord); unityFogFactor = exp2(-unityFogFactor*unityFogFactor)
    
#else
    // 未启用雾效时，雾效因子设为0（完全透明）
    #define UNITY_CALC_FOG_FACTOR_RAW(coord) float unityFogFactor = 0.0
#endif

// 封装雾效计算流程：先通过UNITY_Z_0_FAR_FROM_CLIPSPACE转换深度坐标，再调用原始计算宏
#define UNITY_CALC_FOG_FACTOR(coord) UNITY_CALC_FOG_FACTOR_RAW(UNITY_Z_0_FAR_FROM_CLIPSPACE(coord))

// 定义雾效插值器结构，vectype指定存储类型（如float1），idx为TEXCOORD索引
#define UNITY_FOG_COORDS_PACKED(idx, vectype) vectype fogCoord : TEXCOORD##idx;

代码：使用 Unity 内置方案

Shader "Unlit/Course_09_Fog"
{
    Properties
    {
        _Color("Color",Color) = (1,0,1,1)
    }
    
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog

            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                // 相当于是在雾效开启时，定义一个float类型的变量 fogCoord
                UNITY_FOG_COORDS(1)
            };

            fixed4 _Color;

            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);

                // 传入的 o.vertex 表示裁剪空间下的坐标
                UNITY_TRANSFER_FOG(o,o.vertex);
                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 c = _Color;

                // 第二个参数传入物体的颜色
                UNITY_APPLY_FOG(i.fogCoord,c);
                return c;
            }
            ENDCG
        }
    }
}

概念：雾效的实现 - 方法三

  • 分析：
    • 复用 worldPos.w 存储雾效系数，减少 GPU 插值寄存器占用，对移动端性能友好；
  • 代码：

{
    Properties
    {
        _Color("Color",Color) = (1,0,1,1)
    }
    
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog

            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                // 无需额外的定义雾效插值器，直接将存储世界坐标的worldPos变量的w分量
                float4 worldPos : TEXCOORD0;
            };

            fixed4 _Color;

            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
				o.worldPos.xyz = mul(unity_ObjectToWorld,v.vertex).xyz;
				
				// 计算雾效因子
				UNITY_TRANSFER_FOG_COMBINED_WITH_WORLD_POS(o,o.worldPos);
                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 c = _Color;

				// 这里计算的结果存储在 i.worldPos.w 当中，表示雾效的混合因子
                UNITY_EXTRACT_FOG_FROM_WORLD_POS(i);
                UNITY_APPLY_FOG(_unity_fogCoord,c);
                // 在片段着色器当中，进行雾效颜色混合
                
                return c;
            }
            ENDCG
        }
    }
