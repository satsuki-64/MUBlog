Lecture 3：UV、贴图与溶解效果

UV

UV 的基础概念

概念：UV

  • 概念：什么是 UV
    • UV 是独立于 3D 空间坐标（XYZ）的二维坐标系，用于定位纹理图像上的像素如何贴合到模型表面；
    • 通常 U 代表水平方向，V 代表垂直方向，取值范围通常为 [0,1]；
    • 不管模型有没有 UV，在 Shader 当中都可以写，比如现在 appdata 当中引入；
  • 补充：UV 的颜色为什么是彩色
    • 假设直接按照颜色，输出 UV.x：
      • 在 U 方向上，最左边表示 0 点，因此为黑色。最右边为 1 点，因此为白色；
      • 而中间的所有点，都是根据 0 点和 1 点插值得到的信息；
      • 并且因为输出的是 UV.x，因此计算每个片元时，从上到下的 UV.x 都是同一个值；
    • 假设直接按照颜色，输出 UV.y
    • 如果将 UV 的 x 和 y 值分别放到 RGB 的 R 和 G 通道时，此时得到的颜色：
      • 左下角为 (0,0)，因此就是纯黑色；
      • 右下角为 (1,0)，因此 RGB 为 (1,0,0)，因此即表示红色；
      • 左上角为 (0,1)，因此 RGB 为 (0,1,0)，因此即表示绿色；
  • 示例：
Shader "taecg/Character Unlit"
{
    Properties
    {
		_Tex("Tex",2D) = "white"{}
    }
    
    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _Tex;
	        
            struct appdata
            {
				float4 vertex : POSITION;
				
				// 定义UV
				float4 uv : TEXCOORD0;
            };
            
            // 在v2f的部分只有SV_POSITION是需要指定语义，其他的都不强制具体的含义 
            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
            };
            
            v2f vert (appdata v)
            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				o.uv = v.uv;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            fixed4 tex = tex2D(_Tex,i.uv);
				return tex;
            }
            ENDCG
        }
    }
}

UV 与多层渐变效果

概念：UV 与多层渐变效果 - 方法一

  • 概念：frac 函数
    • frac(x) 返回输入值 x 的小数部分（即 x - floor(x)），结果始终在 [0, 1) 范围内；
    • 比如：frac(3.7) 返回 0.7，frac(-2.3) 返回 0.7（因为 -2.3 - floor(-3) = 0.7）；
  • 分析：UV 与 frac 函数
    • 通过将 UV 坐标乘以系数后取小数，可创建网格、条纹等重复结构；
    • 例如：frac(uv.x * 5) 会将 UV 水平分割为 5 个重复区间；
    • 如果进一步结合 sin 或 dot 函数生成伪随机数，用于模拟自然纹理（如云层、噪点）；

纹理管线

基础概念

概念：纹理管线

  • 概念：
    • 纹理管线主要包含三个关键步骤：投影函数、映射函数和值变换函数；
      1. 投影函数：
      • 将 3D 模型顶点坐标从模型空间转换到 2D 纹理空间；
      • 通过投影矩阵将三维物体表面点 (x,y,z) 映射到二维参数空间 (u,v)；
      1. 映射函数：
      • 黑白纹理图 → 纹理像素坐标 (0.85, 0.82)；
      • 将投影得到的 (u, v) 参数坐标映射到纹理图像的具体像素位置，处理纹理坐标的缩放、平铺和偏移，解决纹理坐标超出 [0,1] 范围的情况（通过 Wrap Mode）；
      • 分析：超出 [0,1] 范围的情况
        • （1）Repeat 模式：重复采样；
        • （2）Clamp 模式：使用最后一个像素来重复；
      1. 值变换函数：
      • 人物局部特写 (435,419) → 着色后图像 (0.58,0.32,0.22)；
      • 将从纹理中采样得到的原始值，转换为最终着色使用的颜色值；
      • 其可能包含了从 8 位纹理值 (0-255) 到浮点颜色值 (0.0-1.0) 的标准化转换；
  • 概念：
    • 这三部当中，可以定制的部分主要就是映射函数的部分，即 UV 采样贴图的操作；

基于纹理贴图进行采样

代码：贴图采样

  • 步骤：
      1. 声明属性：_Name("Display Name",2D) = ""{}；
      1. 定义 CG 语句：sampler2D _Name；
      1. 采样：tex2D(_Name,uv)；
  • 注意：
    • 属性名和 CG 语句里面的变量名必须一一对应，不然 Unity 无法正确传输数据；

Shader "taecg/Character Unlit"
{
    Properties
    {
		_Tex("Tex",2D) = "white"{}
    }
    
    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _Tex;
	        
	        // 只有这个输入端对于语义的要求很严格，需要具体的指定需要的数据来源
            struct appdata
            {
				float4 vertex : POSITION;
				// 表示第一套UV，如果不加数字TEXCOORD，则默认表示第一套 UV
				float4 uv : TEXCOORD0;
            };
            
            // 在v2f的部分，uv的数据不具体指定语义也能正常执行
            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
            };

            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				o.uv = v.uv;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            // 第二个参数是顶点的UV信息
	            // 现在是对_Tex变量上进行采样，但是是进行的固定采样，即采样同一个UV点
	            // fixed4 tex = tex2D(_Tex,0.2);
	            
	            
	            // 现在是正确的对UV进行采样
	            fixed4 tex = tex2D(_Tex,i.uv);
				return tex;
            }
            ENDCG
        }
    }

溶解效果

溶解效果的实现：基于 Clip 裁剪

概念：溶解效果的实现

  • 概念：
    • clip() 是 Unity Shader 编程中 CG/HLSL 语言提供的一个关键函数，它的主要作用是根据条件丢弃当前处理的片元​；
    • 当传递给 clip() 的参数值为负数时，当前片元会被直接丢弃，不参与后续的渲染计算和写入帧缓冲区；
  • 定义：
  • 特点：
      1. 相比手动编写 if-discard 逻辑，clip() 是 GPU 硬件优化的指令，因此优先使用 clip 函数而不是 if 语句；
      1. 可以基于任何计算值决定是否丢弃片元，如纹理采样值、顶点属性、世界坐标等；
  • 分析：基于 Clip 实现溶解效果
    • 溶解 (Dissolve) 效果是游戏中常见的特效，用于表现物体逐渐消失或出现的过程；
      1. 阈值控制​：定义一个从 0 到 1 变化的溶解阈值 (_DissolveThreshold) ；
      1. 噪声采样​：使用噪声纹理为每个片元分配一个随机值 (通常采样噪声纹理的 R 通道)；
      1. 片元丢弃​：当片元的随机值小于当前阈值时，使用 clip() 丢弃该片元；
  • 补充：不同的噪声纹理
    • 云状噪声 (Cloud Noise)：在 Photoshop 中使用"云彩"滤镜生成；
    • 柏林噪声 (Perlin Noise)：程序化生成更自然的渐变；
    • 瓦砾噪声 (Cellular Noise)：产生更破碎的效果；
  • 补充：
    • 和 UV 相关的数据建议使用 float4 格式存储；
  • 代码：
Shader "taecg/Character Unlit"
{
    Properties
    {
	    [Header(Base)]
	    // 加上 [NoScaleOffset] 特性后，当前贴图的 Till 不可被更改 
		[NoScaleOffset]_MainTex("MainTex",2D) = "white"{};
		_Color("Color",Color) = {0,0,0,0};
		
		[Space(10)]
		[Header(DissolveTex)]
		_Clip("Clip",Range(0,1)) = 0;
		
		// 这里的DissolveTex是一张噪声贴图
		_DissolveTex("_DissolveTex",2D) = "white"{};
    }
    
    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _MainTex;
            
            // 后背带“ST”的这种方式声明的变量，存贮的是贴图的 Till 值
            sampler2D _DissolveTex;float _DissolveTex_ST; 
            fixed4 _Color;
            fixed4 _Clip;
            
	        // 只有这个输入端对于语义的要求很严格，需要具体的指定需要的数据来源
            struct appdata
            {
				float4 vertex : POSITION;
				// 表示第一套 UV，如果不加数字 TEXCOORD，则默认表示第一套 UV
				float4 uv : TEXCOORD0;
            };
            
            // 在v2f的部分，uv的数据不具体指定语义也能正常执行
            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
            };
            
            v2f vert (appdata v)
            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				o.uv = v.uv;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            fixed4 c;
	            
	            // tex变量存储当前片元对应的贴图的颜色
	            fixed4 tex = tex2D(_Tex,i.uv);
	            
	            // 叠加颜色
	            c = tex;
	            c += _Color;
		        
		        fixed4 dis = tex2D(_DissolveTex,i.uv)
		        
	            // 将当前贴图的RGB中的R值去减去Clip值，如果当前片元的R-_Clip小于0，则当前片元被裁减
	            clip(dis.r-_Clip);
	            
	            // 返回裁剪后的贴图
				return tex;
            }
            
            ENDCG
        }
    }
}
补充：discard 关键字
  • discard 是片元着色器中的关键字/指令，当丢弃条件非常复杂，无法用简单的数值比较表达，或者你需要更明确的流程控制时，可以使用 discard 来丢弃当前片元；

纹理的 ST 参数：控制 Till 和 Offset

概念：纹理的 ST 参数

  • 概念：
    • 在 Unity 的 Shader 编程中，sampler2D _DissolveTex; float _DissolveTex_ST; 这组变量的第二个成员 _DissolveTex_ST 是一个纹理缩放平移参数，它的作用是为对应的 _DissolveTex 纹理提供 UV 坐标变换的控制能力；
    • ST 是 Scale（缩放）​和 Translation（平移）​的缩写，由 4 个浮点数组成。在 Shader 中声明为 float _DissolveTex_ST 时，Unity 会自动将其扩展为 float4 类型，并通过 Material 面板的纹理属性传递数据；
    • 具体来讲，x 的值代表了其在 u 方向上采样的值；
  • 分析：平铺控制
    • 通过 _DissolveTex_ST.xy 调整噪声纹理的平铺密度：
      • i.uv = i.uv * _DissolveTex_ST.xy; // 缩放UV坐标
    • 因此，若 _DissolveTex_ST.xy = (2, 2)，则噪声纹理会在模型表面重复 2 次，溶解效果会更细碎；
  • 分析：平移控制
    • 通过 _DissolveTex_ST.zw 移动纹理采样起点；
      • i.uv = i.uv + _DissolveTex_ST.zw; // 偏移UV坐标
    • 可用于动态调整溶解起始位置（如通过脚本修改偏移量实现溶解动画）；
  • 图示：

分析：在片元着色器当中进行 Till 和 Offset 控制

  • 代码：
Shader "taecg/Character Unlit"
{
    Properties
    {

    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _MainTex;
            sampler2D _DissolveTex;float _DissolveTex_ST;
            fixed4 _Color;
            fixed4 _Clip;

            struct appdata
            {
				float4 vertex : POSITION;
				float4 uv : TEXCOORD0;
            };

            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
            };
            
            v2f vert (appdata v)
            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				o.uv = v.uv;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            fixed4 c;
	            fixed4 tex = tex2D(_Tex,i.uv);
	            c = tex;
	            c += _Color;
		        
		        // * _DissolveTex_ST.x：更改当前噪声纹理的平铺密度
		        // + _DissolveTex_ST.zw：设置当前纹理的 Offset
		        fixed4 dis = tex2D(_DissolveTex, i.uv * _DissolveTex_ST.xy + _DissolveTex_ST.zw)
	            
	            clip(dis.r-_Clip);
				return tex;
            }
            ENDCG
        }

}

分析：优化 - 在顶点着色器当中进行 Till 和 Offset 控制

  • 分析：
    • 顶点着色器对每个顶点执行一次，而片元着色器对每个像素/片元执行一次。一个模型可能有几十个顶点，但可能覆盖成百上千个片元，将操作移至顶点着色器可减少计算量；
    • 例如，一个覆盖 1000 个像素的三角形，在片元着色器中进行 Tiling/Offset 计算需要执行 1000 次，而在顶点着色器中只需执行 3 次（每个顶点一次）；
    • 并且，顶点着色器输出的变量会通过光栅化阶段自动在三角形表面进行插值，UV 坐标的 Tiling/Offset 结果可以通过插值平滑过渡到每个片元，无需每个片元单独计算；
  • 代码：
Shader "taecg/Character Unlit"
{
    Properties
    {
	    [Header(Base)]
		[NoScaleOffset]_MainTex("MainTex",2D) = "white"{};
		_Color("Color",Color) = {0,0,0,0};
		
		[Space(10)]
		[Header(DissolveTex)]
		_Clip("Clip",Range(0,1)) = 0;
		_DissolveTex("_DissolveTex",2D) = "white"{};
    }
    
    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _MainTex;
            sampler2D _DissolveTex;float _DissolveTex_ST; // 这种方式声明的变量（后背带ST的），存贮的是Till值
            fixed4 _Color;
            fixed4 _Clip;

            struct appdata
            {
				float4 vertex : POSITION;
				float4 uv : TEXCOORD0;
            };

            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
				float4 dissolveUV : TEXCOORD1;
            };
            
            v2f vert (appdata v)
            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				// 错误方法：会连着主纹理和噪声纹理一起更改，效果错误，因此需要将主纹理和噪声纹理的UV拆分
				// o.uv.xw = v.uv*_DissolveTex_ST.xy + _DissolveTex_ST.zw;
				
				// 正确方法：将噪声的纹理分开来进行处理，因此需要在v2f中单独存储噪声纹理部分的UV
				o.uv = v.uv;
				o.dissolveUV.xy = v.uv * _DissolveTex_ST.xy + _DissolveTex_ST.zw;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            fixed4 c;
	            fixed4 tex = tex2D(_Tex,i.uv);
	            c = tex;
	            c += _Color;
	            
	            // 单独使用噪声的UV
		        fixed4 dis = tex2D(_DissolveTex,i.dissolveUV)
	            clip(dis.r-_Clip);
				return tex;
            }
            ENDCG
        }
    }

优化溶解效果

分析：溶解的原理

  • 分析：噪声的纹理
    • 噪声：
    • 目标的效果：
      • 噪声纹理的中间呈现颜色的梯度变化，越靠近中心点、颜色越明亮；
      • 此种效果可以使用一张专门的颜色渐变贴图来表示；
    • 渐变的效果：
  • 分析：
    • 当噪声贴图上、对应 UV 位置的数值为 0 时（即黑色），则对 RampTex 图上采样的位置也为 0（高亮）；
    • 如果对应 UV 位置的数值为 1 时（即白色），则对 RampTex 图上采样的位置也为 1（即采样黑色，因此值为 0，因此得到模型原始的颜色，因为这里是使用的加法、来将 RampTex 上的颜色加上来，因此就是原本颜色）；
  • 补充：方法 - smoothstep(min,max,x)
    • 如果 x 比 min 小，返回 0；
    • 如果 x 比 max 大，返回 1；
    • 如果 x 处于范围 [min，max] 中，则返回 0 和 1 之间的平滑值；
    • 不过 smoothstep 的计算量比较大，其是平滑插值，有比较多的乘除运算，因此能不用尽量不使用，比如可以考虑用 saturate（线性插值）代替；
  • 补充：方法 - saturate(x)
    • 如果 x 比 0 小，返回 0；
    • 如果 x 比 1 大，返回 1；
    • 如果 x 在 0-1之间，则返回 x；
  • 代码：
Shader "taecg/Character Unlit"
{
    Properties
    {
	    [Header(Base)]
		[NoScaleOffset]_MainTex("MainTex",2D) = "white"{};
		_Color("Color",Color) = {0,0,0,0};

		[Space(10)]
		[Header(DissolveTex)]
		_Clip("Clip",Range(0,1)) = 0;
		[NoScaleOffset]_DissolveTex("_DissolveTex",2D) = "white"{};
		_RampTex("RampTex(RGB)",2D) = "white"{}
    }
    
    SubShader
    {        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            sampler2D _MainTex;
            sampler2D _DissolveTex;float _DissolveTex_ST;
            
            // 虽然这里的高亮贴图是二维的，但是因为只会使用上一维的信息，因此可以使用1D来进行存储
            sampler _RampTex;
            
            fixed4 _Color;
            fixed4 _Clip;

            struct appdata
            {
				float4 vertex : POSITION;
				float4 uv : TEXCOORD0;
            };

            struct v2f
            {
				float4 pos : SV_POSITION;
				float4 uv : TEXCOORD0;
            };
            
            v2f vert (appdata v)
            {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				o.uv.xy = v.uv.xy;
				o.uv.zw = v.uv * _DissolveTex_ST.xy + _DissolveTex_ST.zw;
				return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
	            fixed4 c;
	            fixed4 tex = tex2D(_Tex,i.uv.xy);
	            c = tex;
	            c += _Color;
	            
	            // 单独使用噪声的UV
		        fixed4 dis = tex2D(_DissolveTex,i.uv.zw)
	            clip(dis.r-_Clip);
	            
	            // fixed4 rampTex = tex2D(_RampTex,i.uv);
	            // 仅仅只是将贴图的效果加了上来，而没有选择采样点：c += rampTex;
		        
		        // 将当前噪声贴图的RGB值的R值作为插值系数，得到0~1之间的值，假设当前R=0.5，则当前返回红色的效果
		        // 并且因为这里的 min = Clip，因此当Clip变化时、当前插值的结果也会不一样，其使得当前溶解的边缘（处于Clip值附近的UV点）始终处于高亮
		        // 然后设置 max = Clip + 1，其目的是保证当前如果Clip=0时，此时其他地方不会叠加上RampTex纹理的效果
		        
		        fixed dissolveValue = saturate(dissolveTex.r - _Clip)/(_Clip + 0.1 - _Clip); 
		        fixed4 rampTex = tex1D(_RampTex,dissolveValue);
		        C += rampTex;
				return tex;
            }
            ENDCG
        }
    }
}
