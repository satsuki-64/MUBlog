透射效果实现

分析：计算

  • 代码：基于 dot(-L,V) 计算透射效果
half4 frag (Varyings i) : SV_Target
{
	half4 color;
	half4 baseMap = SAMPLE_TEXTURE2D(_BaseMap,sampler_BaseMap,i.uv);
	color = baseMap * _BaseColor;
	color.rgb = MixFog(color.rgb,i.fogCoord);

	// --------- 简单的漫反射效果 ----------
	float3 N = normalize(i.normalWS);
	Light mainLight = GetMainLight();
	float3 L = mainLight.direction;
	half NdotL = max(0.2,dot(N,L));

	// --------- 透射 -----------
	half3 V = i.viewWS;
	half _LdotV = dot(-L,V);
	
	return _LdotV;
}

边缘透射偏转

概念：边缘透射偏转

  • 概念：
    • 边缘透射偏转（Edge Transmission Bias）的核心目标是通过调整光线在物体内部的传播路径，模拟复杂几何体（如半透明雕像、树叶等）的厚度变化对透射效果的影响；
  • 实现：
// --------- 简单的漫反射效果 ----------
float3 N = normalize(i.normalWS);
Light mainLight = GetMainLight();
float3 L = mainLight.direction;
half NdotL = max(0.2,dot(N,L));
color = NdotL;

// --------- 透射 -----------
half3 V = i.viewWS;

// --------- 透射偏转效果 ----------
half3 H = L + N * _NormalDestortion;
half _LdotV = dot(-H,V);
  • 分析：
    • 原本的 _LdotV 是使用 dot(_L,V) 计算得到，用于表示透射之后的能量值。但是现在使用 -H，而 H = L + N * _NormalDestortion，其含义表示从 L 到 N 的一个半角向量：
    • 因此其表示的是当光线穿过物体之后、其在物体内部的一个偏移，而 _NormalDestortion 用来控制这个偏移的程度。当 _NormalDestortion 越小，越是接近无偏移，则公式退化为简单的透视效果实现，即 dot(_L,V)，但如果 _NormalDestortion 变得更大，则更加偏向于偏移的视角，偏移的效果越强；

透射的强度与衰减

概念：透射的强度与衰减

  • 分析：
    • 使用 pow(saturate(_LdotV),_Attenuation) * _Strength 来引入透射的强度与衰减，然后引入 mainLight.color；
  • 代码：
// --------- 简单的漫反射效果 ----------
float3 N = normalize(i.normalWS);
Light mainLight = GetMainLight();
float3 L = mainLight.direction;
half NdotL = max(0.2,dot(N,L));
color = NdotL;

// --------- 透射 -----------
half3 V = i.viewWS;
half3 H = L + N * _NormalDestortion;
half _LdotV = dot(-H,V);
half3 I = pow(saturate(_LdotV),_Attenuation) * _Strength;
I *= mainLight.color;

color.rgb *= I;
color.rgb += I;

材质的透射厚度

概念：材质的透射厚度

  • 目的：
    • 当前物体上的不同区域，它们可能拥有不同的厚度，比如如下图所示：
    • 因此，对于不同厚度的区域，它们应具有不同的透射强度。对于厚度比较薄的区域，透射的效果更加的强烈，而对于厚度更厚的地方，则透视的效果更弱，这就需要实现材质的透射厚度；
  • 实现：材质的透射厚度
    • （1）Invert the surface normals - 反转表面的法线
    • （2）Render ambient occlusion - 渲染 AO 图
      • 因为 AO 图可以用于增强局部的阴影效果，因此可以类似于比较薄的区域；
    • （3）Invert the colors and store in texture
      • 反转 AO 图的颜色，并将其存储在贴图当中；
      • 对此贴图采样，然后用于乘以透射的颜色，即可控制透射的效果强度；
  • 代码：
half4 frag(Varyings i) : SV_Target
{
    half4 c;
    half4 baseMap = SAMPLE_TEXTURE2D(_BaseMap, sampler_BaseMap, i.uv);
    c = baseMap * _BaseColor;

    // ---------------- 漫反射 -------------------
    float3 N = normalize(i.normalWS);
    Light mainLight = GetMainLight();
    float3 L = mainLight.direction;
    half NdotL = max(0.3, dot(N, L));
    c *= NdotL;

	// ----------------- 厚度与透射 --------------------
    half thicknessMap = 1 - SAMPLE_TEXTURE2D(_ThicknessMap, sampler_ThicknessMap, i.uv).r;
    c.rgb += LightingTranslucent(L, i.viewWS, N, mainLight.color, thicknessMap); // 在内部，最终输出的结果乘以 thicknessMap 的值

    // ----------------- 额外光的透射支持 ----------
    #ifdef _ADDITIONAL_LIGHTS
    uint pixelLightCount = GetAdditionalLightsCount();
    for (uint lightIndex = 0u; lightIndex < pixelLightCount; ++lightIndex)
    {
        Light light = GetAdditionalLight(lightIndex, i.positionWS);
        half3 attenuatedLightColor = light.color * (light.distanceAttenuation * light.shadowAttenuation);
        c.rgb += LightingTranslucent(light.direction, i.viewWS, N, light.color, thicknessMap) * attenuatedLightColor;
    }
    #endif

    c.rgb = MixFog(c.rgb, i.fogCoord);
    return c;
}
