方法一：基于法线进行顶点偏移

原理

概念：描边基础概念

  • 描边需要两个 Pass ，当中第一个 Pass 就正常进行绘制，比如光照、纹理等等；
  • 然后第二个 Pass 需要生成纯色，为其填充其他的颜色，然后将此模型放大；
    • 这里的放大不是一般的放大，而是让其沿着顶点在法线上偏移；
  • 然后此时两个 Pass 的绘制如下

Shader "Unlit/Cartoon"
{
    Properties
    {
        _BaseColor("BaseColor",Color) = (1,1,1,1)
        _BaseMap("BaseMap",2D) = "white"{}
        
        [Header(Outline)]
        _OutlineWidth("OutlineWidth",Range(0,1)) = 0.1
        _OutlineColor("OutlineColor",Color) = (0,0,0,1)
    }
    
    SubShader
    {
        Tags
        {
            "RenderType"="Opaque"
            "RenderPipeline" = "UniversalPipeline"
            "IgnoreProjector" = "True"
        }
        LOD 100

        Pass
        {
            Tags
            {
                "LightMode" = "UniversalForward"
            }
            
            Name "CartoonLit"
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            #include "Packages/com.unity.render-pipelines.core/ShaderLibrary/Color.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Lighting.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/ShaderGraphFunctions.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                float2 uv : TEXCOORD0;
            };
            
            CBUFFER_START(UnityPerMaterial)
                half4 _BaseColor;
            CBUFFER_END
            TEXTURE2D(_BaseMap);SAMPLER(sampler_BaseMap);float4 _BaseMap_ST;

            Varyings vert (Attributes v)
            {
                Varyings o;
                o.positionCS = TransformObjectToHClip(v.positionOS);
                o.uv = TRANSFORM_TEX(v.uv,_BaseMap);

                return o;
            }

            half4 frag (Varyings i) : SV_Target
            {
                half4 color;
                half4 baseMap = SAMPLE_TEXTURE2D(_BaseMap,sampler_BaseMap,i.uv);
                color = baseMap * _BaseColor;
                
                return color;
            }
            
            ENDHLSL
        }

        Pass
        {
            Tags
            {
                "LightMode" = "SRPDefaultUnlit"
            }
            Cull Front
            
            Name "CartoonVertexOffset"
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            #include "Packages/com.unity.render-pipelines.core/ShaderLibrary/Color.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Lighting.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float3 normalOS : NORMAL;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
            };

			// 变量定义在 CBuffer 当中
            CBUFFER_START(UnityPerMaterial)
                half4 _BaseColor;
                half _OutlineWidth;
                half4 _OutlineColor;
            CBUFFER_END
            TEXTURE2D(_BaseMap);SAMPLER(sampler_BaseMap);float4 _BaseMap_ST;

            Varyings vert (Attributes v)
            {
                Varyings o;
                float3 positionOS = v.positionOS;
                positionOS += normalize(v.normalOS) * _OutlineWidth;
                
                o.positionCS = TransformObjectToHClip(positionOS);
                return o;
            }

            half4 frag (Varyings i) : SV_Target
            {
                return _OutlineColor;
            }
            
            ENDHLSL
        }
    }
}

方法二：外轮廓描边

原理

分析：外轮廓描边

  • 方式一：使用深度
    • 在第一个绘制描边颜色的外扩的 Pass 当中，不写入物体的深度值。然后在第二个 Pass 当中写入深度值，因此第二个 Pass 的颜色就会替换掉第一个 Pass 的颜色；
    • 问题：会产生大量的 Overdraw，因为第一个 Pass 只是用来绘制一圈描边，它整个画面渲染出来之后、除了描边的区域、其他地方都会被第二个 Pass 所覆盖，因此产生大量的没必要的 Overdraw；
  • 方法二：模板测试
    • 先在第一个 Pass 中正常绘制，然后同时往模板缓冲区中写入一个模板值，此会导致所有人物模型的位置具有同样的模板值；
    • 然后再在第二个绘制描边的 Pass 中进行比较；
    • Pass 1
    • Pass 2
    • 新问题：当前物体具有很多个子物体，因此不同物体按照不同次序渲染时，会导致问题；
    • 解决：
      • 先完成所有的物体的第一个 Pass，然后再执行所有物体的第二个 Pass；
      • 这里可以使用 RenderObject 技术；

描边随视距变换

基础概念

概念：描边随视距变换

  • 分析：
    • 现在的描边效果，会受到近大远小的影响，因此走进物体时，会看到“更粗”的描边大小，但是可能会希望、无论举例物体多近，描边大小都是一样的，因此需要一些额外的设计；
  • 思路：
    • 让当前描边的宽度乘以一个数值，当物体距离相机越来越远的时候，乘以的这个数值越来越大，是的这个描边的宽度不变；

实现

概念：描边随视距变换实现

  • 实现：
    • 注意这里的宽度是三维向量；
    • 计算相机到当前顶点的世界坐标下距离，然后使用此距离乘以宽度，即可得到正确的效果；
  • 分析：
    • 现在的问题是距离远了之后、可能描边过大；
  • 实现：
    • 使用一个 lerp 来对 distance 进行插值；

完整代码

Shader "Unlit/Cartoon"
{
    Properties
    {
        _BaseColor("BaseColor",Color) = (1,1,1,1)
        _BaseMap("BaseMap",2D) = "white"{}
        
        [Header(Outline)]
        _OutlineWidth("OutlineWidth",Range(0,1)) = 0.1
        _OutlineColor("OutlineColor",Color) = (0,0,0,1)
        _UnifomWidth("UnifomWidth",Range(0,1)) = 0.5
    }
    
    SubShader
    {
        Tags
        {
            "RenderType"="Opaque"
            "RenderPipeline" = "UniversalPipeline"
            "IgnoreProjector" = "True"
        }
        LOD 100

		// 第一个Pass就是正常的采样纹理贴图，然后进行渲染
        Pass
        {
            Tags
            {
                "LightMode" = "UniversalForward"
            }
            
            Stencil
            {
                Ref 1
                Comp Always
                Pass Replace
            }
            
            Name "CartoonLit"
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            #include "Packages/com.unity.render-pipelines.core/ShaderLibrary/Color.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Lighting.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/ShaderGraphFunctions.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                float2 uv : TEXCOORD0;
            };
            
            CBUFFER_START(UnityPerMaterial)
                half4 _BaseColor;
            CBUFFER_END
            TEXTURE2D(_BaseMap);SAMPLER(sampler_BaseMap);float4 _BaseMap_ST;

            Varyings vert (Attributes v)
            {
                Varyings o;
                o.positionCS = TransformObjectToHClip(v.positionOS);
                o.uv = TRANSFORM_TEX(v.uv,_BaseMap);

                return o;
            }

            half4 frag (Varyings i) : SV_Target
            {
                half4 color;
                half4 baseMap = SAMPLE_TEXTURE2D(_BaseMap,sampler_BaseMap,i.uv);
                color = baseMap * _BaseColor;
                
                return color;
            }
            
            ENDHLSL
        }

		// 描边的实现在第二个Pass
        Pass
        {
            Tags
            {
                "LightMode" = "SRPDefaultUnlit1"
            }
            
            Stencil
            {
                Ref 1
                Comp NotEqual
            }
            
            Cull Front
            
            Name "CartoonVertexOffset"
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            #include "Packages/com.unity.render-pipelines.core/ShaderLibrary/Color.hlsl"
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Lighting.hlsl"

            struct Attributes
            {
                float4 positionOS : POSITION;
                float3 normalOS : NORMAL;
                // 这个属性可以在外部的DCC软件当中，为其赋值
                half4 color : COLOR;
            };

            struct Varyings
            {
                float4 positionCS : SV_POSITION;
                half4 color : TEXCOORD0;
            };

			// 变量定义在 CBuffer 当中
            CBUFFER_START(UnityPerMaterial)
                half4 _BaseColor;
                half _OutlineWidth;
                half4 _OutlineColor;
                half _UnifomWidth;
            CBUFFER_END
            TEXTURE2D(_BaseMap);SAMPLER(sampler_BaseMap);float4 _BaseMap_ST;

            Varyings vert (Attributes v)
            {
                Varyings o;
                
                // 思路：描边的宽度本身不会变化，只是由于我们离的远了，所以感觉是变细了，因此只需要对描边宽度乘上一个越来越大的值做为弥补即可
                // 求出相机和顶点间的距离
                float3 positionWS = TransformObjectToWorld(v.positionOS);
                // 当 distance = 1 时，就表示正常的进大远小的表现
                float distance = length(_WorldSpaceCameraPos - positionWS);
                distance = lerp(1,distance,_UnifomWidth);
                
                float3 positionOS = v.positionOS;
                float3 width = normalize(v.normalOS) * _OutlineWidth;
                width *= distance;
                width *= v.color.a;
                positionOS += width;
                
                o.positionCS = TransformObjectToHClip(positionOS);
                
                o.color = v.color;
                return o;
            }

            half4 frag (Varyings i) : SV_Target
            {
                return half4(_OutlineColor.rgb * i.color.rgb,1);
            }
            
            ENDHLSL
        }
    }
}
