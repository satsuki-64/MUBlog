实现分析：MVVM 在 WPF 中的实现

分析：实现分析 - MVVM 在 WPF 中的实现

  • 在 WPF 中，你会像如下这样去定义一个专门管理视图 View 的 ViewModel：
using System.ComponentModel;

/// <summary>
/// 歌曲的视图模型（ViewModel），作为 View 和 Model 之间的桥梁。
/// 负责暴露数据、处理视图逻辑，并通过实现 INotifyPropertyChanged 接口实现数据绑定通知。
/// </summary>
public class SongViewModel : INotifyPropertyChanged
{
    #region Construction
    /// <summary>
    /// 构造 SongViewModel 的默认实例。
    /// 初始化时创建一个默认的 Song 模型对象，其艺术家名称和歌曲标题均为 "Unknown"。
    /// 此构造函数通常由视图（XAML）或设置 DataContext 的代码调用。
    /// </summary>
    public SongViewModel()
    {
        _song = new Song { ArtistName = "Unknown", SongTitle = "Unknown" };
    }
    #endregion

    #region Members
    /// <summary>
    /// 后台字段，存储 Song 模型对象的引用。
    /// 视图模型（ViewModel）包装此模型（Model），为其添加数据绑定支持。
    /// </summary>
    Song _song;
    #endregion

    #region Properties
    /// <summary>
    /// 获取或设置视图模型所包装的 Song 模型对象。
    /// 注意：直接设置此属性不会自动触发属性变更通知（PropertyChanged）。
    /// 如果需要在模型整体更换时更新视图，应考虑在此属性的 set 访问器中调用 RaisePropertyChanged。
    /// </summary>
    public Song Song
    {
        get
        {
            return _song;
        }
        set
        {
            _song = value;
        }
    }

    /// <summary>
    /// 获取或设置歌曲的艺术家名称。
    /// 此属性直接映射到底层 Song 模型的 ArtistName 属性。
    /// 当 setter 检测到值实际发生变化时，会调用 RaisePropertyChanged 方法，
    /// 触发 PropertyChanged 事件，通知所有绑定的视图控件（如 TextBlock）更新显示。
    /// 这是实现数据双向绑定的关键环节
    /// </summary>
    public string ArtistName
    {
        get { return Song.ArtistName; } // 从模型获取值
        set
        {
            // 检查新值是否与当前值不同，避免不必要的更新[1](@ref)
            if (Song.ArtistName != value)
            {
                // 将新值设置到底层模型
                Song.ArtistName = value;
                // 触发属性变更通知，名字为 "ArtistName"
                // 绑定到该属性的UI元素将自动更新[1,6](@ref)
                RaisePropertyChanged("ArtistName");
            }
        }
    }
    #endregion

    #region INotifyPropertyChanged Members
    /// <summary>
    /// 属性变更事件。
    /// 当视图模型的某个属性值发生变化时触发。
    /// WPF 的数据绑定引擎会自动监听此事件，并在事件触发时更新绑定的UI元素
    /// </summary>
    public event PropertyChangedEventHandler PropertyChanged;
    #endregion

    #region Methods
    /// <summary>
    /// 引发 PropertyChanged 事件的辅助方法。
    /// 封装了事件触发逻辑，确保线程安全（通过获取事件委托的本地副本）
    /// </summary>
    /// <param name="propertyName">发生变化的属性名称。应与属性访问器中传递的名称一致。</param>
    private void RaisePropertyChanged(string propertyName)
    {
        // 获取事件委托的本地副本，防止多线程环境下在检查 null 后事件被取消订阅而引发的空引用异常
        PropertyChangedEventHandler handler = PropertyChanged;
        // 检查是否有事件订阅者
        if (handler != null)
        {
            // 触发事件，传递发送者（this）和包含属性名称的事件参数
            handler(this, new PropertyChangedEventArgs(propertyName));
        }
    }
    #endregion
}
  • 同时在 View 中你需要使用 Binding 将 ViewModel 的属性绑定和控件的内容相绑定：
<TextBox Content="{Binding ArtistName}" />
  • 注意：
    • 要实现 View 和 ViewModel 双向绑定，我们的 ViewModel 必须实现 INotifyPropertyChanged 接口，由于 WPF Framework 让控件监听了 PropertyChanged 事件，当属性值发生时，触发 PropertyChanged 事件，所以控件就能自动获取到最新的值；
    • 反之，当控件的值发生改变时，例如 TextBox 触发 OnTextChanged 事件，自动将最新的值同步到 ViewModel 相应的属性中；
  • 总结：
    • ViewModel 提供了 View 需要的数据，并且 ViewModel 实现 INotifyPropertyChanged 接口 ，当数据更改时，触发了 PropertyChanged 事件，由于控件也监听了此事件，在事件的响应函数里实现数据的更新；

Unity 中实现 MVVM 的不同

分析：Unity 中实现 MVVM 的不同

  • 分析：
    • Unity 3D 没有提供数据绑定，可以参考之前 MVP 的设计理念：将 UI 抽象成独立的一个个 View，将面向 Component 开发转换为面向 View 开发，每一个 View 都有独立的 ViewModel 进行管理；
    • 由于 Unity 3D 没有 XAML，也没有 Data Binding 技术，故只能在抽象出来的 View 中去实现类似于 WPF 的 Data Binding，Converter，Command 等；
  • 注意：MVVM 实现数据绑定与 MVP 的 Presenter 的区别
    • MVP 设计模式中数据的绑定是通过将具体的 View 实例传递到 Presenter 中完成的；
    • MVVM 是以数据改变之后、引发对应的事件，在事件完成数据更新的；

MVVM 的核心交互模式

概念：MVVM 的核心交互模式

  • 概念：
    • MVVM 架构的成功，很大程度上依赖于以下几种规范的交互方式，它们是实现解耦的基石；
    • （1）数据绑定 (Data Binding)：
      • View 通过声明式的绑定表达式（如 {Binding UserName}）自动从 ViewModel 获取数据并展示，当 ViewModel 中的数据变化时，View 会自动更新，无需手动操作 UI 控件。理想情况下，View 应该是被动的，它只负责显示绑定过来的数据； 
    • （2）命令模式 (Command Pattern)：
      • 用户交互操作（如按钮点击）不应在 View 的后台代码中处理，而应绑定到 ViewModel 上实现。这确保了业务逻辑与 UI 层完全分离，View 只负责触发命令，而具体做什么由 ViewModel 决定；
    • （3）观察者模式/消息机制 (Observer Pattern/Messaging)：
      • 对于更复杂的交互，比如导航、弹出对话框或跨 ViewModel 通信，可以使用消息机制（如 MVVM Light Toolkit 中的 Messenger）
      • ViewModel 发送消息，其他 ViewModel 订阅并响应这些消息，从而避免直接的引用依赖；

// 定义一个专门用来管理 View 的 ViewModel，它以属性的形式提供数据，以方法的形式提供行为
// ViewModel 中的属性是特殊的属性，它必须具备当数据更改时通知订阅者这个功能
// 怎么通知订阅者？当然是事件，故我把此属性称为 BindableProperty 属性
public class BindableProperty<T>
{
    public delegate void ValueChangedHandler(T oldValue, T newValue);

    public ValueChangedHandler OnValueChanged;

    private T _value;
    public T Value
    {
        get
        {
            return _value;
        }
        set
        {
            if (!object.Equals(_value, value))
            {
                T old = _value;
                _value = value;
                // 当值发生变化时，会自动调用 OnValueChanged 当中存储的若干回调
                ValueChanged(old, _value);
            }
        }
    }

    private void ValueChanged(T oldValue, T newValue)
    {
        if (OnValueChanged != null)
        {
            OnValueChanged(oldValue, newValue);
        }
    }

    public override string ToString()
    {
        return (Value != null ? Value.ToString() : "null");
    }
}

// 定义一个通用的 View，将它命名为 ViewBase，所有的自定义 View 需要继承此 ViewBase
public abstract class ViewBase<T> : MonoBehaviour, IView<T> where T : ViewModelBase
{
	private bool _isInitialized;

	#region 将 View 和 ViewModel 使用 BindableProperty 进行绑定  
	// ViewModelProperty 表示当前 View 绑定的 ViewModel
	// 补充：在这里使用 BindableProperty 的原因，是为了当 View 绑定的 ViewModel 发生更改时、会自动触发对应回调
    public readonly BindableProperty<ViewModel> ViewModelProperty = new BindableProperty<ViewModel>();
    
    // BindingContext 表示当前 View 与 ViewModel 绑定的上下文属性
    // 当为一个 View 的 BindingContext 指定 ViewModelBase 时，初始化时势必会触发 OnValueChanged 事件
    public T BindingContext
    {
        get 
        { 
	        // 返回的内容为当前 View 绑定的 ViewModel
	        return ViewModelProperty.Value; 
		}
        set 
        { 
	        // 如果当前 View 未初始化，则在第一次更改时执行初始化，调用 OnInitialize 方法
	        if (!_isInitialized)
			{
				OnInitialize();
				_isInitialized = true;
			}
			
			// 如果更改当前 View 绑定的 ViewModel，会触发对应的 OnValueChanged 回调
			ViewModelProperty.Value = value;
		}
    }
	#endregion

	/// <summary>
	/// 初始化View，当 BindingContext 改变时执行
	/// </summary>
	protected virtual void OnInitialize()
	{
		//无所ViewModel的Value怎样变化，只对OnValueChanged事件监听(绑定)一次
		ViewModelProperty.OnValueChanged += OnBindingContextChanged;
	}

	// 当绑定的上下文（ViewModel）发生变化时，执行此代码进行解绑
    public virtual void OnBindingContextChanged(ViewModel oldViewModel, ViewModel newViewModel)
    {
	    // 后续使用 属性绑定器 来实现解绑和重新绑定
    }
}

分析：Unity 实现 MVVM 的架构

分析：Unity 实现 MVVM 的架构

  • 分析：
  • 概念：Model
    • 代表应用程序的核心数据，如 Song 类，包含 ArtistName、SongTitle 等数据属性，是纯 C# 类；
  • 概念：ViewModel
    • 作为 View 和 Model 之间的桥梁，将 Model 数据转换为 View 可直接使用的格式；
    • 通过 BindableProperty<T> 实现数据绑定能力；
    • 其替代了传统 MVP 模式中 Presenter 的主动更新，采用事件驱动的被动更新模式；
  • 概念：View
    • 负责 UI 显示和用户交互处理，通过 ViewBase 基类统一管理 ViewModel 绑定；
  • 分析：数据流动
    • Model 数据变化 → ViewModel 属性更新 → 触发 OnValueChanged 事件 → View 接收事件并更新 UI；
    • 用户操作 UI → View 捕获交互事件 → 更新 ViewModel 属性 → 可选地同步到 Model 层；
  • 总结：
    • 通过以上代码实现了 MVVM 代码的基础框架，接下来为 View 上的变量实现属性绑定器；

属性绑定器

属性绑定器的概念及其实现

概念：属性绑定器

  • 概念：
    • PropertyBinder 的核心作用是自动化地建立 ViewModel 中可绑定属性，即 BindableProperty<T> 与 View 中 UI 更新方法之间的订阅关系，它充当了数据驱动视图的“粘合剂”，是 MVVM 模式中实现双向绑定的关键基础设施；
  • 实现：
      1. 注册绑定：
      • View 告诉 PropertyBinder，请帮我监听 ViewModel 中名为 X 的属性，当它变化时，请调用 Y 方法；
      1. 记录操作：
      • PropertyBinder 利用反射获取目标属性的信息，并将“订阅”和“取消订阅”这两个操作分别封装成委托，存入对应的列表中；
      1. 执行绑定与解绑：
      • 当 View 的 BindingContext（即其对应的 ViewModel）被设置或切换时，PropertyBinder 会遍历所有已记录的“绑定”或“解绑”操作，批量执行，从而高效地管理所有属性的订阅关系；

添加绑定

分析：添加绑定

  • 代码：PropertyBinder.cs 添加绑定的代码实现
/// <summary>
/// 将变量和指定的回调进行绑定
/// </summary>
/// <param name="name">具体 ViewModel 中的属性名</param>
/// <param name="valueChangedHandler">属性待绑定的回调</param>
/// <typeparam name="TProperty">待绑定变量的类型</typeparam>
public void Add<TProperty>(string name,BindableProperty<TProperty>.ValueChangedHandler valueChangedHandler )
{
	// fieldInfo 是一个 FieldInfo 类型，存储了一个指定了泛型类型的 BindableProperty<T> 类型
	// 因为 BindableProperty 存储在具体 ViewModel 当中，因此利用反射，在具体 ViewModel 中获取指定变量名的变量
	var fieldInfo = typeof(T).GetField(name, BindingFlags.Instance | BindingFlags.Public);
	
	// 如果当前 T 对应的 ViewModel 当中没有指定的属性
	if (fieldInfo == null)
	{
		throw new Exception(string.Format("Unable to find bindableproperty field '{0}.{1}'", typeof(T).Name, name));
	}

	// 在 _binders 链表当中添加一个委托，
	// viewmodel 是一个形式参数, 它声明了这个Lambda表达式需要接收一个类型为 T（即 ViewModelBase）的参数
	_binders.Add(viewmodel =>
	{
		GetBindablePropertyFromViewModel<TProperty>(name, viewmodel, fieldInfo).OnValueChanged += valueChangedHandler;
	});

	_unbinders.Add(viewModel =>
	{
		GetBindablePropertyFromViewModel<TProperty>(name, viewModel, fieldInfo).OnValueChanged -= valueChangedHandler;
	});
}

/// <summary>
/// 找到指定变量变量的 BindableProperty 属性，进而可以获得其 OnValueChanged 变量
/// </summary>
/// <param name="name">变量名</param>
/// <param name="viewModel">具体 ViewModel 类型的实例</param>
/// <param name="fieldInfo">存储了指定了 BindableProperty 泛型类型的FieldInfo</param>
/// <typeparam name="TProperty"></typeparam>
/// <returns>变量对应的BindableProperty</returns>
private BindableProperty<TProperty> GetBindablePropertyFromViewModel<TProperty>(string name, T viewModel,FieldInfo fieldInfo)
{
	// 从 viewModelName这个对象实例中，获取由 fieldInfo 所代表的那个字段的当前值
	// 使用当前传入的具体 ViewModel 实例，获取用于绑定具体 ViewModel 的 BindableProperty 属性
	var value = fieldInfo.GetValue(viewModel);
	
	BindableProperty<TProperty> bindableProperty = value as BindableProperty<TProperty>;
	if (bindableProperty == null)
	{
		throw new Exception(string.Format("Illegal bindableproperty field '{0}.{1}' ", typeof(T).Name, name));
	}

	return bindableProperty;
}

PropertyBinder 完整代码

using System;
using System.Collections.Generic;
using System.Reflection;

namespace SubProject.UIFramework.MVVM
{
    /// <summary>
    /// 属性绑定器，用于管理 View 和 ViewModel 之间属性的自动绑定与解绑。
    /// 它通过反射动态订阅 ViewModel 中 BindableProperty 的 OnValueChanged 事件，实现数据到视图的自动通知。
    /// </summary>
    /// <typeparam name="T">具体的 ViewModel 类型，必须继承自 ViewModelBase</typeparam>
    public class PropertyBinder<T> where T : ViewModelBase
    {
        // 委托定义：用于封装绑定和解绑一个具体 ViewModel 的操作
        private delegate void BindHandler(T viewmodel);
        private delegate void UnBindHandler(T viewmodel);

        // 存储所有绑定操作的委托列表
        private readonly List<BindHandler> _binders = new List<BindHandler>();
        // 存储所有解绑操作的委托列表，与_binders中的项一一对应
        private readonly List<UnBindHandler> _unbinders = new List<UnBindHandler>();

        /// <summary>
        /// 添加一个属性绑定
        /// 当调用 Bind(T) 方法时，将为指定的 ViewModel 属性订阅值改变事件。
        /// </summary>
        /// <typeparam name="TProperty">可绑定属性的数据类型</typeparam>
        /// <param name="name">ViewModel 中 BindableProperty 字段的名称</param>
        /// <param name="valueChangedHandler">当属性值发生变化时需要执行的委托方法</param>
        /// <exception cref="Exception">当通过反射找不到指定名称的字段时抛出</exception>
        public void Add<TProperty>(string name, BindableProperty<TProperty>.ValueChangedHandler valueChangedHandler)
        {
            // 1. 使用反射获取指定名称的字段信息。
            // BindingFlags.Instance | BindingFlags.Public 确保只查找公共的实例字段。
            var fieldInfo = typeof(T).GetField(name, BindingFlags.Instance | BindingFlags.Public);

            if (fieldInfo == null)
            {
                throw new Exception(string.Format("Unable to find bindableproperty field '{0}.{1}'", typeof(T).Name, name));
            }

            // 2. 向绑定列表中添加一个委托，该委托的功能是：为某个 ViewModel 的特定属性订阅值改变事件。
            _binders.Add((T viewmodel) =>
            {
                // 获取目标 ViewModel 实例中的 BindableProperty 对象
                var bindableProperty = GetPropertyValue<TProperty>(name, viewmodel, fieldInfo);
                // 将 UI 更新的处理方法订阅到该属性的值改变事件上
                bindableProperty.OnValueChanged += valueChangedHandler;
            });

            // 3. 向解绑列表中添加一个对应的委托，该委托的功能是：取消订阅。
            // 这是确保在 ViewModel 切换或 View 销毁时能够正确清理订阅，防止内存泄漏的关键。
            _unbinders.Add((T viewModel) =>
            {
                var bindableProperty = GetPropertyValue<TProperty>(name, viewModel, fieldInfo);
                bindableProperty.OnValueChanged -= valueChangedHandler;
            });
        }

        /// <summary>
        /// 通过反射从 ViewModel 实例中获取指定字段的 BindableProperty 值，并进行类型安全检查。
        /// </summary>
        /// <typeparam name="TProperty">期望的属性数据类型</typeparam>
        /// <param name="name">字段名（用于错误信息）</param>
        /// <param name="viewModel">ViewModel 实例</param>
        /// <param name="fieldInfo">通过反射获取的字段信息</param>
        /// <returns>有效的 BindableProperty<TProperty> 对象</returns>
        /// <exception cref="Exception">当获取的对象不是 BindableProperty<TProperty> 类型时抛出</exception>
        private BindableProperty<TProperty> GetPropertyValue<TProperty>(string name, T viewModel, FieldInfo fieldInfo)
        {
            var value = fieldInfo.GetValue(viewModel);
            BindableProperty<TProperty> bindableProperty = value as BindableProperty<TProperty>;

            if (bindableProperty == null)
            {
                throw new Exception(string.Format("Illegal bindableproperty field '{0}.{1}' ", typeof(T).Name, name));
            }

            return bindableProperty;
        }

        /// <summary>
        /// 为指定的 ViewModel 执行所有已注册的绑定操作。
        /// 通常在 View 的 BindingContext 被设置为新的 ViewModel 后调用。
        /// </summary>
        /// <param name="viewmodel">要绑定的目标 ViewModel</param>
        public void Bind(T viewmodel)
        {
            if (viewmodel != null)
            {
                // 遍历所有绑定委托，并应用到这个 ViewModel 上
                for (int i = 0; i < _binders.Count; i++)
                {
                    _binders[i](viewmodel);
                }
            }
        }

        /// <summary>
        /// 为指定的 ViewModel 执行所有已注册的解绑操作。
        /// 通常在 View 的 BindingContext 被切换或 View 被销毁前调用，以防止内存泄漏。
        /// </summary>
        /// <param name="viewmodel">要解绑的目标 ViewModel</param>
        public void Unbind(T viewmodel)
        {
            if (viewmodel != null)
            {
                // 遍历所有解绑委托，并从这 ViewModel 上取消订阅
                for (int i = 0; i < _unbinders.Count; i++)
                {
                    _unbinders[i](viewmodel);
                }
            }
        }
    }
}
