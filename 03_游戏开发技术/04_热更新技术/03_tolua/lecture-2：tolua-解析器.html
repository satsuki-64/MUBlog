toLua 解析器

代码

  • 代码：C# 脚本
using UnityEngine;
// toLua提供的命名控件
using LuaInterface;

public class LuaStateTest : MonoBehaviour
{
	private void Start()
	{
		LuaState luaState = new LuaState();
		
		// 启动解析器（虚拟机）
		luaState.Start();
		
		// 使用toLua执行Lua代码
		luaState.DoString("print('你好世界')");
		
		// 另一个重载
		// 执行Lua重载，并且指明出处，方便调试时查看问题
		luaState.DoString("print('你好世界')","LuaStateTest.cs");
		
		// 执行Lua文件
		// 方式一
		luaState.DoFile("Main.lua");
		
		// 方式二
		// 第二种执行Lua文件的方式不需要加.lua后缀
		luaState.Require("Main");
		
		// 检查解析器的栈顶是否置空
		luaState.CheckTop();
		
		// 销毁解析器
		luaState.Dispose();
		luaState = null;
	}
}
  • 代码：Lua 脚本
print("你好")
总结：关于 LuaState
    1. LuaState 是引擎，负责管理整个 Lua 环境和 C# 交互的桥梁；
    1. 用 DoString 执行简单片段或测试；
    1. 用 DoFile ​需要谨慎，通常仅在需要强制重新加载文件时使用；
    1. 用 Require 进行模块化加载，这是大多数情况下优先使用的方法；
    1. 销毁解析器时，先调用 CheckTop，再调用 Dispose；

自定义解析路径

概念：自定义解析路径

  • 概念：
    • 通过 toLua 的解析路径，自定义 toLua 的解析路径；
    • ToLua 默认的 Lua 脚本搜索路径通常是 Assets/Lua 目录，此时分为两种情况：
    • （1）如果该文件属于 Lua 文件夹下那么可以直接加父目录；
    • （2）要让 ToLua 找到其他位置的脚本，最基本的方法是使用 AddSearchPath 来添加解析路径；
  • 代码：
LuaState luaState = new LuaState();
luaState.Start();

// 方式一：如果该文件属于Lua文件夹下，那么可以直接加父目录
luaState.Require("CSharpCallLua/TestLoader");

// 方式二：使用AddSearchPath添加路径
luaState.AddSearchPath(Application.dataPath + "/Lua/CSharpCallLua");
luaState.Require("TestLoader"); // 添加搜索路径后，可以直接执行
注意：关于解析路径和解析方式
  • 如果使用方式一，则 Lua 文件必须得位于 Lua 文件夹下，Lua 同级的其他文件夹下的 Lua 代码不能执行；
  • 和 XLua 中不同，ToLua 中不能和 XLua 的 AddLoader 一样、直接自定义解析函数，需要使用其他方式；

自定义解析器

自定义解析器流程解析

概念：自定义解析器

  • 目标：
    • 通过 toLua 的解析器，自定义解析方式；
  • 总结：
    • 要实现自定义 Lua 解析器，只需要重写 LuaFileUtils 文件中的 ReadFile 方法；
  • 流程：Require 方法的执行流程
    [list2node]
  • 自定义解析器流程
    • 第一步：执行解析器的 Require 方法
      • 进入方法体，并执行到 LuaRequire 方法
    • 第二步：进入 LuaRequire 方法
      • 通过 LuaDLL.tolua_requre，调用静态方法 Loader；
    • 第三步：进入静态方法 Loader
      • 在这其中进入一个很重要的类 LuaFileUtils，此类用于自定义 Lua 脚本的加载方式；
    • 第四步：进入 LuaFileUtils 类的 ReadFile 方法
      • 在 ReadFile 方法当中执行 FindFile 方法，得到文件的路径；
      • 然后使用 File.ReadAllBytes 方法读取 Lua 文件的所有的字节数组；
    • 第五步：执行 FindFile 方法
      • FindFile 方法中的核心代码就是一个 for 循环，其循环条件是一个 searchPaths 的 List 链表；
      • 只要往 searchPaths 的 List 链表中传入一个名字，那么 ToLua 在执行的时候就会多遍历一个路径；

自定义 Lua 文件解析器代码实现

代码：自定义 Lua 文件解析器代码实现

using UnityEngine;
using LuaInterface;

public class LuaCustomLoader : LuaFileUtils
{
	public override byte[] ReadFile(string fileName)
	{
		// 使用LuaState之前，可以先new一个自定义的解析器
		// 因为在LuaFileUtils的构造函数时，其会返回示例，并调用ReadFile的方法
		
		// 如果没有Lua后缀，则先加上 .lua 后缀
		// 不管从 AB 包中加载还是从 res 下加载 都不支持用 .1ua 后缀 所以 toLua 加上了 bytes 后缀
		if(!fileName.EndWith(".lua"))
		{
			fileName += ".lua";
		}
		byte[] buffer = null;
		
		// 进行Lua热更新的lua代码，主要是自己写的lua逻辑，而ToLua自带的lua代码都在Resources当中，直接从recourse下进行加载即可
		// 第二种：从AB包中加载
		// 由于AB包加载的方法只需要文件名，因此拆分
		string[] strs = fileName.Split('/');
		// 加载AB包的Lua文件
		TextAsset luaCode = ABMgr.GetInstance().LoadRes<TextAsset>("lua",str[strs.Length-1]);
		if(luaCode!=null)
		{
			buffer = luaCode.bytes;
			Resources.UnloadAsset(luaCode);
		}
		
		// 如果第二种方法没有成功加载，则使用第一种方法
		if(buffer == null)
		{
			// 第一种：从 Resources 中加载 lua 文件
			// toLua中自带一个 Copy Lua Files to Resources 的功能，其只能将Lua文件夹下的lua文件复制到Resources下
			// 每次打包之前只需要点一次Copy Lua File to Resources即可打包到Resources中
			string path = "Lua/" + fileName;
			TextAsset text = Resources.Load<TextAsset>(path);
			if(text != null)
			{
				buffer = text.bytes;
				
				// 卸载使用后的文本资源
				Resources.UnloadAsset(text);
			}
		}
		
		return buffer;
	}
}
分析：为什么 LuaCustomLoader 类解析了之后就可以使用
  • 之所以 LuaCustomLoader 类解析了之后就可以使用，是因为父类是一个单例对象，并且父类的构造函数当中，初始化了 instance 变量；
  • 如果想要重新定义解析 Lua 的方式，只需要在 ReadFile 方法中进行逻辑编写；
总结：关于自定义解析路径
    1. 通过继承 LuaFileUtils 并且重写 ReadFile 函数，达到一个自定义解析方式
    1. 需要在使用 LuaState 之前，new 出来自定义文件加载类；
    1. 具体的加载方式，第二种时 Resources 文件夹下面，一种是 AB 包中；
    1. Resources 文件夹中时：
    • （1）先点击 Copy Lua Files to Resources；
    • （2）在通过代码来加载；
    1. AB 包中时：
    • （1）先打 AB 包；
    • （2）再通过代码来加载；
    1. 完整的解析器代码，先尝试加载 AB 包中的内容，再在为空的情况下、加载 Resources 下的内容；

toLua 解析器管理器

概念：toLua 解析器管理器

  • 目的：
    • 通过 toLua 解析器管理器，来管理唯一的一个 toLua 管理器；
    • 将此管理器，写成一个继承自 MonoBehaviour 的单例，其具有执行、销毁、初始化的方法；
    • 外部使用时，都是通过 LuaMgr 管理器来统一执行；
  • 代码：LuaMgr.cs
public class LuaMgr : SingletonAutoMono<LuaMgr>
{
	private LuaState luaState;

	public void Init()
	{
		// 实例化自己定义的、继承自`LuaFileUtils`文件的自定义解析类
		// 最后打包的时候再取消注释
		// new LuaCustomLoader();
		
		// 初始化唯一的 luaState
		luaState = new LuaState();
		luaState.Start(); 
		
		// 委托初始化
		// 想要C#和Lua相互访问使用委托，必须要在初始化时把委托工厂初始化了，不然没法使用委托
		DelegateFactory.Init();
		
		// 协程相关
		// 如果要让ToLua的协程运行起来，必须要添加一个脚本
		LuaLooper loop = this.gameObject.AddComponent<LuaLooper>();
		// 将自己声明的解析器和luaLooper绑定起来，这样就可以让协程跑起来
		loop.luaState = luaState;		
		
		// Lua协程注册
		LuaCoroutline.Register(luaState,this);
		
		// Lua使用Unity中的类相关
		LuaBinder.Bind(luaState);
	}
	
	public LuaState LusState
	{
		get
		{
			return luaState;	
		}
	}
	
	public void DoString(string str,string chunkName = "LuaMgr.cs")
	{
		// 默认使用"LuaMgr.cs"作为执行引擎
		if(luaState!=null)
		{
			luaState.DoString(str);
		}
	}
	
	// 执行指定的Lua脚本名
	public void Require(string fileName)
	{
		luaState.Require(fileName);
	}
	
	// 销毁管理器
	public void Dispose()
	{
		if(luaState == null)
		{
			return;
		}
		
		luaState.CheckTop();
		luaState.Dispose();
		luaState= null;
	}
}
