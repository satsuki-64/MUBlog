全局变量获取

概念：全局变量获取

  • 代码：Main.lua
-- 全局变量
testNumber = 1        -- 声明了一个名为 testNumber 的全局变量，并赋值为数字 1
testBool = true       -- 声明了一个名为 testBool 的全局变量，并赋值为布尔值 true
testFloat = 1.2       -- 声明了一个名为 testFloat 的全局变量，并赋值为浮点数 1.2
testString = "123"    -- 声明了一个名为 testString 的全局变量，并赋值为字符串 "123"

local testLocal = 10  -- 声明了一个名为 testLocal 的局部变量，并赋值为数字 10
  • 代码：
public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		// 执行"Main.lua"代码
		LuaMgr.GetInstance().Require("Main");
		
		// 1. 获取全局变量
		// 直接通过LuaState的索引器，在中括号内填入变量名，即可得到全局变量
		Debug.Log(LuaMgr.GetInstance().LuaState["testNumber"]);
		
		// 2. 修改值
		LuaMgr.GetInstance().LuaState["testNumber"] = 99;
		Debug.Log(LuaMgr.GetInstance().LuaState["testNumber"]); // 打印出来99
		
		// 3. 增加全局变量
		LuaMgr.GetInstance().LuaState["addNumber"] = 100;
	}
}
总结：关于 ToLua 中变量获取
    1. toLua 中也没有办法通过 C# 得到 toLua 中明的局部临时变量；
    1. 使用 LuaState["testNumber"] 得到的变量是 Object 类型，因此需要类型转换，比如：
    • int value = Convert.ToInt32(LuaMgr.GetInstance().LuaState["testNumber"]);
    1. 可以在 C# 处，为 lua 中添加全局变量，相当于在 Lua 的 _G 中添加了全局变量；

无参无返回函数的获取

概念：无参无返回函数的获取

  • 代码：
-- 全局变量
testNumber = 1        -- 声明了一个名为 testNumber 的全局变量，并赋值为数字 1
testBool = true       -- 声明了一个名为 testBool 的全局变量，并赋值为布尔值 true
testFloat = 1.2       -- 声明了一个名为 testFloat 的全局变量，并赋值为浮点数 1.2
testString = "123"    -- 声明了一个名为 testString 的全局变量，并赋值为字符串 "123"

testFun = function()
	print("无参无返回")
end
  • 代码：
using UnityEnige;
using LuaInterface;

public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		// 方式一：使用 GetFunction 获取
		LuaFunction f1 = LuaMgr.GetInstance().GetFunction("testFun");
		// 调用无参无返回函数
		f1.Call();
		// 执行完成之后，记得销毁
		f1.Dispose();
		
		// 方法二：通过 中括号 函数名的形式
		// 此方法和获得全局变量的方式一致
		f1 = LuaMgr.GetInstance().LuaState["testFun"] as LuaFunction;
		// 调用无参无返回函数
		f1.Call();
		// 执行完成之后，记得销毁
		f1.Dispose();
	
		// 方式三：转为委托使用
		// 这种方式需要先得到一个 LuaFunction，然后再多写一步、将其写成委托存储起来
		f1 = LuaMgr.GetInstance().LuaState["testFun"] as LuaFunction;
		// 通过LuaFunction中的ToDelegate方法，将其存储到委托当中
		UnityAction action = f1.ToDelegate<UnityAction>();
		// 调用
		action();
	}
}
注意：Lua 函数的委托初始化
  • 想要 C# 和 Lua 相互访问使用委托，必须要在初始化时把委托工厂初始化了，不然没法使用委托；
  • 加入此代码：DelegateFactory.Init();

有参有返回函数的获取

概念：有参有返回函数的获取

  • 代码：
-- 全局变量
testNumber = 1        -- 声明了一个名为 testNumber 的全局变量，并赋值为数字 1
testBool = true       -- 声明了一个名为 testBool 的全局变量，并赋值为布尔值 true
testFloat = 1.2       -- 声明了一个名为 testFloat 的全局变量，并赋值为浮点数 1.2
testString = "123"    -- 声明了一个名为 testString 的全局变量，并赋值为字符串 "123"

testFun = function()
	print("无参无返回")
end

testFun2 = function(a)
	print("有餐有返回函数")
	return a + 100
end
  • 代码：
using UnityEnige;
using LuaInterface;

public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		// 方式一：通过 LuaFunction的 Call 来获得
		f2 = LuaMgr.GetInstance().GetFunction("testFun2");
		// 1. 开始使用
		f2.BeginPCall();
		// 2. 传参数
		f2.Push(99);
		// 3. 结束传参数并调用
		f2.PCall();
		// 4. 得到返回值
		int result = (int)f2.CheckNumber();
		// 5. 执行结束
		f2.EndPCall();
		
		// 方式二：通过 LuaFunction 的 Invoke 方法来调用
		// 其中前面的参数为传入参数的类型，最后的参数为返回值类型
		// 小括号内填入传参的内容
		result = f2.Invoke<int,int>(199);
		
		// 方式三：转委托
		Func<int,int> func = f2.ToDelegate<Func<int,int>>();
		result = func(900);
		
		// 方式四：不得function，直接执行
		// 参数一表示函数名，参数二表示传入的参数，参数三默认填写true即可
		result = LuaMgr.GetInstance().LuaState.Invoke<int,int>("testFun2",900,true);
	}
}

// 注意：第一个参数是正常的传参，函数的返回值是第一个返回值，之后的out标识的都是返回值
public delegate int CustomCallOut(int a,out int b,out bool c,out string d,out int e)

public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		// 多返回值函数
		// 方式一：通过Call的方式
		function = LuaMgr.GetInstance().LuaState.GetFunction("testFun3");
		// 开始调用
		function.BeginPCall();
		// 传入参数
		function.Push(1000);
		// 传参结束，开始调用
		function.PCall();
		// 得到返回值
		function.PCall();
		// 按照返回值的返回顺序，以此存储返回的内容
		int a1 = (int)function.CheckNumber();
		int b1 = (int)function.CheckNumber();
		bool c1 = function.CheckBoolean();
		string d1 = function.CheckString();
		int e1 = (int)function.CheckNumber();
		
		// 执行结束
		function.EndPCall();
		Debug.Log("Check接的第一个返回值：" + a1);
		
		// 方式二：通过out来接返回值
		CustomCallOut callOut = function.ToDelegate<CustomCallOut>();
		
		int b2;
		bool c2;
		string d2;
		int e2;
		// 第一个返回值通过result得到
		result = callOut(999,out b2,out c2,out d2,out e2);
	}
}

testNumber = 1        -- 声明了一个名为 testNumber 的全局变量，并赋值为数字 1

testFloat = 1.2       -- 声明了一个名为 testFloat 的全局变量，并赋值为浮点数 1.2
testString = "123"    -- 声明了一个名为 testString 的全局变量，并赋值为字符串 "123"

// 使用委托来装载它
using UnityEnige;
using LuaInterface;

// 变长参数的委托
public delegate void CustomCallParams(int a,params object[] objs);

public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		// 多返回值函数
		// 方式一：通过Call的方式
		function = LuaMgr.GetInstance().LuaState.GetFunction("testFun4");
		
		CustomCallParams call = function.ToDelegate<CustomCallParams>();
		call(100,1,true,"123",0,false);
		
		// 方式二：通过在LuaFunction中直接Call
		// 其中的泛型决定传入的参数的类型，然后科研直接执行
		function.Call<int,int,bool,string,int>(10,100,true,"123",0);
	}
}
注意：记得在自定义列表里面加上当前新建的委托
注意：在 LuaFunction 中直接 Call 来调用的问题
  • 比如 function.Call<int,int,bool,string,int>(10,100,true,"123",0); 这种方式，它只是用于无返回值的类型。其可以直接传入参数后使用。如果有返回值，这种方式就无效；

访问 Lua 中的字典

概念：访问 Lua 中的字典

  • 代码：
using UnityEnige;
using LuaInterface;

public class Test2 : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		LuaTable dic = LuaMgr.GetInstance().LuaState.GetTable("testDic");
		Debug.Log(dic["1"]); // 输出结果为999
		LuaDictTable<string,int> luaDic = dic2.ToDectTable<string,in>();
		Debug.Log(luaDic["1"]); // 输出结果为999
		
		LuaTable dic2 = LuaMgr.GetInstance().LuaState.GetTable("testDic2");
		Debug.Log(dic2[true]); // 报错，无法输出
		LuaDictTable<object,object> luaDic2 = dic2.ToDectTable<object,object>();
		Debug.Log(luaDic2[true]); // 可以正确输出
		
		// 迭代器遍历
		IEnumerator<LuaDictEntry<object,object> ie = luaDic2.GetEnumerator();
		while(ie.MoveNext())
		{
			Debug.Log(ie.Current.Key + "_" + ie.Current.Value);
		}
		
	}
}
总结：C# 中获取 Lua 字典
    1. 想要通过中括号得到 Lua 中字典的值，只支持 string 和 int 类型；
    1. 可以使用 LuaDictTable<object,object> 来装载其他类型的字典，并且其泛型的类型可以固定。如果是多种类型，就使用 object；
    1. 如果想要遍历 Lua 字典，建议使用迭代器来遍历；
    1. 只要是使用 LuaTable 获得的对象，都是引用拷贝，无论链表还是字典；

[mindmap]

  • 链表和字典
    • List
      • 遍历：通过 ToArray 转成数组遍历；

使用 ToLua 中提供的协程

概念：使用 ToLua 中提供的协程

  • 代码：
local coDelay = nil

-- 开启协程的函数
-- 这个函数是toLua提供的
StartDelay = function()
	-- 开启协程的逻辑是ToLua提供的
	-- lua自带的是create、resume、yield、status、wrap、running这些内容
	coDelay = coroutine.start(Delay)
end

Delay = function()
	local c = 1
	while true do
		-- 等待1秒，执行一次
		-- 此方法是由toLua提供，帮忙处理的，在Lua中直接使用是无效的
		coroutine.wait(1)
		print("Count"..c)
		c = c + 1
		if c >= 5 then
			StupDelay()
			break
		end
	end
end

StopDelay = function()
	coroutine.stop(coDelay)
end
  • 代码：
using UnityEnige;
using LuaInterface;

// 变长参数的委托
public delegate void CustomCallParams(int a,params object[] objs);

public class Test : MonoBehaviour
{
	private void Start()
	{
		LuaMgr.GetInstance().Init();
		LuaMgr.GetInstance().Require("Main");
		
		LuaFunction function = LuaMgr.GetInstance().LuaState.GetFunction("StartDelay");
		function.Call();
		function.Dispose(); 
	}
}
总结：使用协程
  • C# 调用 toLua 中的协程，其实主要协程逻辑都在 Lua 中 C# 相当于就是执行了一个函数而已；
  • 而且如果想要让 toLua 的协程运行起来，则其必须要加 LuaLooper 脚本，并且和解析器绑定；
